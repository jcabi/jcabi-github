5   package com.jcabi.github;
6   
7   import com.jcabi.aspects.Immutable;
8   import java.io.IOException;
9   
10  /**
11   * GitHub Git Data References.
12   * @see <a href="https://developer.github.com/v3/git/references/">References API</a>
13   * @since 0.8
14   */
15  @Immutable
16  public interface References {
17  
18      /**
19       * Owner of them.
20       * @return Repo
21       */
22      Repo repo();
23  
24      /**
25       * Creates a reference.
26       * @param ref The name of the fully qualified reference (ie: refs/heads/master).
27       * @param sha The SHA1 value to set this reference to.
28       * @return Reference - The newly created Reference
29       * @throws IOException - If there are any errors.
30       */
31      Reference create(
32          String ref,
33          String sha
34      ) throws IOException;
35  
36      /**
37       * Get Reference by identifier.
38       * @param identifier Reference's name.
39       * @return Reference The reference with the given name
40       */
41      Reference get(
42          String identifier
43      );
44  
45      /**
46       * Iterates all references.
47       * @return Iterator of references.
48       */
49      Iterable<Reference> iterate();
50  
51      /**
52       * Iterates references in sub-namespace.
53       * @param subnamespace Sub-namespace
54       * @return Iterator of references.
55       */
56      Iterable<Reference> iterate(
57          String subnamespace
58      );
59  
60      /**
61       * Iterate references under "tags" sub-namespace.
62       * @return Iterator of references.
63       */
64      Iterable<Reference> tags();
65  
66      /**
67       * Iterate references under "heads" sub-namespace.
68       * @return Iterator of references.
69       */
70      Iterable<Reference> heads();
71  
72      /**
73       * Removes a reference by its identifier.
74       * @param identifier Reference's identifier.
75       * @throws IOException If there is any I/O problem.
76       */
77      void remove(
78          String identifier
79      ) throws IOException;
80  }
