30  package com.jcabi.github;
31  
32  import com.jcabi.aspects.Immutable;
33  import com.jcabi.aspects.Loggable;
34  import com.jcabi.http.Request;
35  import com.jcabi.http.response.RestResponse;
36  import java.io.IOException;
37  import java.net.HttpURLConnection;
38  import lombok.EqualsAndHashCode;
39  import org.hamcrest.Matchers;
40  
41  /**
42   * Github starring API.
43   *
44   * @author Paul Polishchuk (ppol@ua.fm)
45   * @version $Id: fd9148e63ecbc74a9fde94d9beb5919f40ad5ff0 $
46   * @since 0.15
47   * @see <a href="https://developer.github.com/v3/activity/starring/">Starring API</a>
48   */
49  @Immutable
50  @Loggable(Loggable.DEBUG)
51  @EqualsAndHashCode(of = {"owner", "request"})
52  final class RtStars implements Stars {
53  
54      /**
55       * RESTful request.
56       */
57      private final transient Request request;
58  
59      /**
60       * Repository.
61       */
62      private final transient Repo owner;
63  
64      /**
65       * Public ctor.
66       * @param req Request
67       * @param repo Repository
68       */
69      RtStars(final Request req, final Repo repo) {
70          final Coordinates coords = repo.coordinates();
71          this.request = req.uri()
72              .path("/user/starred")
73              .path(coords.user())
74              .path(coords.repo())
75              .back();
76          this.owner = repo;
77      }
78  
79      @Override
80      public Repo repo() {
81          return this.owner;
82      }
83  
84      @Override
85      public boolean starred() throws IOException {
86          return this.request
87              .fetch().as(RestResponse.class)
88              .assertStatus(
89                  Matchers.is(
90                      Matchers.oneOf(
91                          HttpURLConnection.HTTP_NO_CONTENT,
92                          HttpURLConnection.HTTP_NOT_FOUND
93                      )
94                  )
95              ).status() == HttpURLConnection.HTTP_NO_CONTENT;
96      }
97  
98      @Override
99      public void star() throws IOException {
100         this.request
101             .method(Request.PUT)
102             .fetch().as(RestResponse.class)
103             .assertStatus(HttpURLConnection.HTTP_NO_CONTENT);
104     }
105 
106     @Override
107     public void unstar() throws IOException {
108         this.request
109             .method(Request.DELETE)
110             .fetch().as(RestResponse.class)
111             .assertStatus(HttpURLConnection.HTTP_NO_CONTENT);
112     }
113 }
