5   package com.jcabi.github;
6   
7   import com.jcabi.aspects.Immutable;
8   import jakarta.json.JsonObject;
9   import java.io.IOException;
10  
11  /**
12   * Markdown API.
13   * @see <a href="https://developer.github.com/v3/markdown/">Markdown API</a>
14   * @since 0.6
15   */
16  @Immutable
17  public interface Markdown {
18  
19      /**
20       * Get its owner.
21       * @return GitHub
22       */
23      GitHub github();
24  
25      /**
26       * Render.
27       * @param json JSON parameters
28       * @return HTML
29       * @throws IOException If it fails due to I/O problem
30       */
31      String render(JsonObject json)
32          throws IOException;
33  
34      /**
35       * Raw rendering.
36       * @param text Text in Markdown format
37       * @return HTML
38       * @throws IOException If it fails due to I/O problem
39       */
40      String raw(String text)
41          throws IOException;
42  
43  }
