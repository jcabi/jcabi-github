5   package com.jcabi.github;
6   
7   import com.jcabi.aspects.Immutable;
8   import com.jcabi.aspects.Loggable;
9   import com.jcabi.http.Request;
10  import jakarta.json.JsonObject;
11  import java.io.IOException;
12  import lombok.EqualsAndHashCode;
13  
14  /**
15   * GitHub Tag.
16   * @since 0.15
17   */
18  @Immutable
19  @Loggable(Loggable.DEBUG)
20  @EqualsAndHashCode(of = {"request", "owner", "sha" })
21  final class RtTag implements Tag {
22  
23      /**
24       * RESTful request.
25       */
26      private final transient Request request;
27  
28      /**
29       * Repository.
30       */
31      private final transient Repo owner;
32  
33      /**
34       * SHA of the tag.
35       */
36      private final transient String sha;
37  
38      /**
39       * Public constructor.
40       * @param req The request.
41       * @param repo The owner repo.
42       * @param key The sha.
43       */
44      RtTag(
45          final Request req,
46          final Repo repo,
47          final String key
48      ) {
49          this.sha = key;
50          this.owner = repo;
51          this.request = req.uri().path("/repos").path(repo.coordinates().user())
52              .path(repo.coordinates().repo()).path("/git").path("/tags")
53              .path(this.sha).back();
54      }
55  
56      @Override
57      public JsonObject json() throws IOException {
58          return new RtJson(this.request).fetch();
59      }
60  
61      @Override
62      public Repo repo() {
63          return this.owner;
64      }
65  
66      @Override
67      public String key() {
68          return this.sha;
69      }
70  
71  }
