144     /**
145      * Smart event with extra features.
146      * @since 0.6.1
147      */
148     @Immutable
149     @ToString
150     @Loggable(Loggable.DEBUG)
151     @EqualsAndHashCode(of = { "event", "jsn" })
152     final class Smart implements Event {
153         /**
154          * Encapsulated event.
155          */
156         private final transient Event event;
157 
158         /**
159          * SmartJson object for convenient JSON parsing.
160          */
161         private final transient SmartJson jsn;
162 
163         /**
164          * Public ctor.
165          * @param evt Event
166          */
167         public Smart(final Event evt) {
168             this.event = evt;
169             this.jsn = new SmartJson(evt);
170         }
171 
172         /**
173          * Does it have an author?
174          * @return TRUE if the author exists
175          * @throws IOException If there is any I/O problem
176          */
177         public boolean hasAuthor() throws IOException {
178             return !this.event.json().isNull("actor");
179         }
180 
181         /**
182          * Get its author.
183          * @return Author of comment
184          * @throws IOException If there is any I/O problem
185          */
186         public User author() throws IOException {
187             return this.event.repo().github().users().get(
188                 this.event.json().getJsonObject("actor").getString("login")
189             );
190         }
191 
192         /**
193          * Get its type.
194          * @return State of issue
195          * @throws IOException If there is any I/O problem
196          */
197         public String type() throws IOException {
198             return this.jsn.text("event");
199         }
200 
201         /**
202          * Get its URL.
203          * @return URL of issue
204          * @throws IOException If there is any I/O problem
205          */
206         public URL url() throws IOException {
207             try {
208                 return new URI(this.jsn.text("url")).toURL();
209             } catch (final URISyntaxException ex) {
210                 throw new IllegalArgumentException(ex);
211             }
212         }
213 
214         /**
215          * When this issue was created.
216          * @return Date of creation
217          * @throws IOException If there is any I/O problem
218          */
219         public Date createdAt() throws IOException {
220             try {
221                 return new GitHub.Time(
222                     this.jsn.text("created_at")
223                 ).date();
224             } catch (final ParseException ex) {
225                 throw new IllegalStateException(ex);
226             }
227         }
228 
229         /**
230          * Label that was added or removed in this event (if any).
231          * @return Label that was added or removed
232          * @throws IOException If there is any I/O problem
233          * @since 0.24
234          */
235         public Optional<Label> label() throws IOException {
236             Optional<Label> lab = Optional.absent();
237             final JsonObject lbl = this.jsn.json().getJsonObject("label");
238             if (lbl != null) {
239                 lab = Optional.of(
240                     this.event.repo()
241                         .labels()
242                         .get(lbl.getString("name"))
243                 );
244             }
245             return lab;
246         }
247 
248         @Override
249         public Repo repo() {
250             return this.event.repo();
251         }
252 
253         @Override
254         public int number() {
255             return this.event.number();
256         }
257 
258         @Override
259         public JsonObject json() throws IOException {
260             return this.event.json();
261         }
262 
263         @Override
264         public int compareTo(final Event obj) {
265             return this.event.compareTo(obj);
266         }
267     }
