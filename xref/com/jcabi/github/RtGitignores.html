5   package com.jcabi.github;
6   
7   import com.jcabi.aspects.Immutable;
8   import com.jcabi.http.Request;
9   import com.jcabi.http.response.JsonResponse;
10  import com.jcabi.http.response.RestResponse;
11  import jakarta.json.JsonString;
12  import jakarta.ws.rs.core.HttpHeaders;
13  import java.io.IOException;
14  import java.net.HttpURLConnection;
15  import java.util.ArrayList;
16  import java.util.List;
17  import lombok.EqualsAndHashCode;
18  
19  /**
20   * GitHub Gitignore.
21   * <p>Defines storage of .gitignore templates
22   *
23   * @since 0.8
24   */
25  @Immutable
26  @EqualsAndHashCode(of = { "ghub", "request" })
27  final class RtGitignores implements Gitignores {
28  
29      /**
30       * GitHub.
31       */
32      private final transient GitHub ghub;
33  
34      /**
35       * RESTful request.
36       */
37      private final transient Request request;
38  
39      /**
40       * Public CTOR.
41       * @param github GitHub
42       */
43      RtGitignores(
44          final GitHub github) {
45          this.ghub = github;
46          this.request = this.github().entry().uri()
47              .path("/gitignore/templates").back();
48      }
49  
50      @Override
51      public GitHub github() {
52          return this.ghub;
53      }
54  
55      @Override
56      public Iterable<String> iterate() throws IOException {
57          final RestResponse response = this.request.fetch()
58              .as(RestResponse.class)
59              .assertStatus(HttpURLConnection.HTTP_OK);
60          final List<JsonString> list = response.as(JsonResponse.class)
61              .json().readArray().getValuesAs(JsonString.class);
62          final List<String> templates = new ArrayList<>(list.size());
63          for (final JsonString value : list) {
64              templates.add(value.getString());
65          }
66          return templates;
67      }
68  
69      @Override
70      public String template(
71          final String name)
72          throws IOException {
73          return this.request.reset(HttpHeaders.ACCEPT)
74              .header(HttpHeaders.ACCEPT, "application/vnd.github.v3.raw")
75              .uri().path(name).back().fetch()
76              .as(RestResponse.class)
77              .assertStatus(HttpURLConnection.HTTP_OK)
78              .body();
79      }
80  }
