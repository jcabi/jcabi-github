5   package com.jcabi.github;
6   
7   /**
8    * Represents a GitHub notifications thread.
9    * @since 1.0
10   */
11  public interface GitHubThread {
12      /**
13       * Marks this thread as read.
14       * @see <a href="https://developer.github.com/v3/activity/notifications/#mark-a-thread-as-read">Mark a thread as read</a>
15       */
16      void markAsRead();
17  
18      /**
19       * Checks, if the current user is subscribed to this thread.
20       * @return Subscription data, if the user is subscribe, null otherwise.
21       * @see <a href="https://developer.github.com/v3/activity/notifications/#get-a-thread-subscription">Get a Thread Subscription</a>
22       */
23      ThreadSubscription getSubscription();
24  
25      /**
26       * Subscribes the user to the thread and/or makes the user ignore the
27       *  thread.
28       * @param subscribe True, if notifications should be received from this
29       *  thread.
30       * @param ignore True, if all notifications should be blocked from this
31       *  thread.
32       * @return Data of the subscription.
33       * @see <a href="https://developer.github.com/v3/activity/notifications/#set-a-thread-subscription">Set a Thread Subscription</a>
34       */
35      ThreadSubscription setSubscription(boolean subscribe,
36          boolean ignore);
37  
38      /**
39       * Unsubscribes the user from this thread.
40       * @see <a href="https://developer.github.com/v3/activity/notifications/#delete-a-thread-subscription">Delete a Thread Subscription</a>
41       */
42      void deleteSubscription();
43  }
