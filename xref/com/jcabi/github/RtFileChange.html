5   package com.jcabi.github;
6   
7   import com.jcabi.aspects.Loggable;
8   import jakarta.json.JsonObject;
9   import lombok.EqualsAndHashCode;
10  import lombok.ToString;
11  
12  /**
13   * File change.
14   * @since 0.24
15   */
16  @Loggable(Loggable.DEBUG)
17  @EqualsAndHashCode(of = "jsn")
18  @ToString
19  final class RtFileChange implements FileChange {
20      /**
21       * Encapsulated JSON object.
22       */
23      private final transient JsonObject jsn;
24  
25      /**
26       * Public ctor.
27       * @param obj File change JSON object
28       */
29      RtFileChange(
30          final JsonObject obj
31      ) {
32          this.jsn = obj;
33      }
34  
35      @Override
36      public JsonObject json() {
37          return this.jsn;
38      }
39  }
