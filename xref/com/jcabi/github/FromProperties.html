5   package com.jcabi.github;
6   
7   import java.io.IOException;
8   import java.util.Properties;
9   
10  /**
11   * User agent data read from the jcabigithub.properties file.
12   * @since 0.37
13   */
14  @SuppressWarnings("PMD.ConstructorShouldDoInitialization")
15  public final class FromProperties implements UserAgent {
16  
17      /**
18       * Build timestamp.
19       */
20      private static final String JCABI_DATE = "JCabi-Date";
21  
22      /**
23       * Project version.
24       */
25      private static final String JCABI_VERSION = "JCabi-Version";
26  
27      /**
28       * Build number.
29       */
30      private static final String JCABI_BUILD = "JCabi-Build";
31  
32      /**
33       * Properties.
34       */
35      private final transient Properties props = new Properties();
36  
37      /**
38       * Name of the properties file to load.
39       */
40      private final transient String name;
41  
42      /**
43       * Ctor.
44       * @param filename Name of the properties file to look for
45       */
46      public FromProperties(final String filename) {
47          this.name = filename;
48      }
49  
50      @Override
51      public String format() {
52          try {
53              this.props.load(
54                  Thread.currentThread().getContextClassLoader()
55                      .getResourceAsStream(this.name)
56              );
57          } catch (final IOException ex) {
58              throw new IllegalStateException(
59                  String.format("IOException when loading %s", this.name),
60                  ex
61              );
62          }
63          return String.format(
64              "jcabi-github %s %s %s",
65              this.props.getProperty(FromProperties.JCABI_VERSION),
66              this.props.getProperty(FromProperties.JCABI_BUILD),
67              this.props.getProperty(FromProperties.JCABI_DATE)
68          );
69      }
70  }
