5   package com.jcabi.github.safe;
6   
7   import com.google.common.collect.Iterables;
8   import com.jcabi.aspects.Immutable;
9   import com.jcabi.aspects.Loggable;
10  import com.jcabi.github.Comment;
11  import com.jcabi.github.Comments;
12  import com.jcabi.github.Issue;
13  import com.jcabi.github.mock.MkGitHub;
14  import com.jcabi.log.Logger;
15  import java.io.IOException;
16  import java.util.Date;
17  import lombok.EqualsAndHashCode;
18  
19  /**
20   * Safe comments.
21   *
22   * @since 0.34
23   */
24  @Immutable
25  @Loggable(Loggable.DEBUG)
26  @EqualsAndHashCode(of = "origin")
27  public final class SfComments implements Comments {
28  
29      /**
30       * Original comments.
31       */
32      private final transient Comments origin;
33  
34      /**
35       * Public ctor.
36       * @param cmt The original comment
37       */
38      public SfComments(final Comments cmt) {
39          this.origin = cmt;
40      }
41  
42      @Override
43      public String toString() {
44          return this.origin.toString();
45      }
46  
47      @Override
48      public Issue issue() {
49          return this.origin.issue();
50      }
51  
52      @Override
53      public Comment get(final long number) {
54          return new SfComment(this.origin.get(number));
55      }
56  
57      @Override
58      public Iterable<Comment> iterate(final Date since) {
59          return Iterables.transform(
60              this.origin.iterate(since),
61              input -> new SfComment(input)
62          );
63      }
64  
65      @Override
66      public Comment post(final String text) throws IOException {
67          Comment cmt;
68          try {
69              cmt = this.origin.post(text);
70          } catch (final AssertionError ex) {
71              Logger.warn(this, "Failed to post to GitHub: %[exception]s", ex);
72              cmt = new MkGitHub().randomRepo()
73                  .issues().create("", "")
74                  .comments().post(text);
75          }
76          return cmt;
77      }
78  }
