5   package com.jcabi.github;
6   
7   import com.jcabi.aspects.Immutable;
8   import jakarta.json.JsonObject;
9   import java.io.IOException;
10  
11  /**
12   * GitHub Git Data Commits.
13   * @see <a href="https://developer.github.com/v3/git/commits/">Commits API</a>
14   * @since 0.8
15   */
16  @Immutable
17  public interface Commits {
18  
19      /**
20       * Owner of them.
21       * @return Repo
22       */
23      Repo repo();
24  
25      /**
26       * Create a Commit object.
27       * @param params The input for creating the Tag.
28       * @return Commit
29       * @throws IOException - If anything goes wrong.
30       */
31      Commit create(JsonObject params) throws IOException;
32  
33      /**
34       * Return a Commit by its SHA.
35       * @param sha The sha of the Commit.
36       * @return Commit
37       */
38      Commit get(String sha);
39  
40      /**
41       * Return a Statuses object for a given ref (sha, branch name, etc).
42       * @param ref The ref of the Commit
43       * @return Status
44       * @since 0.23
45       */
46      Statuses statuses(String ref);
47  }
