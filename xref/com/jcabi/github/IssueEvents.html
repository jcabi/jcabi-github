5   package com.jcabi.github;
6   
7   import com.jcabi.aspects.Immutable;
8   
9   /**
10   * GitHub issue events.
11   * @see <a href="https://developer.github.com/v3/issues/events/">Issue Events API</a>
12   * @since 0.23
13   */
14  @Immutable
15  public interface IssueEvents {
16      /**
17       * Owner of them.
18       * @return Repo
19       */
20      Repo repo();
21  
22      /**
23       * Get specific issue event by number.
24       * @param number Issue event number
25       * @return Event
26       * @see <a href="https://developer.github.com/v3/issues/events/#get-a-single-event">Get a single event</a>
27       */
28      Event get(int number);
29  
30      /**
31       * Iterate over all issue events.
32       * @return Iterator of issue events
33       * @see <a href="https://developer.github.com/v3/issues/events/#list-events-for-a-repository">List events for a repository</a>
34       */
35      Iterable<Event> iterate();
36  }
