30  package com.jcabi.github;
31  
32  import com.jcabi.aspects.Immutable;
33  import com.jcabi.aspects.Loggable;
34  import com.jcabi.http.Request;
35  import java.io.IOException;
36  import javax.json.JsonObject;
37  import lombok.EqualsAndHashCode;
38  
39  /**
40   * Github Blob.
41   *
42   * @author Alexander Lukashevich (sanai56967@gmail.com)
43   * @version $Id: fd266d8948d3b101b27c3815630e4f6460cddba0 $
44   */
45  @Immutable
46  @Loggable(Loggable.DEBUG)
47  @EqualsAndHashCode(of = {"request", "hash" })
48  final class RtBlob implements Blob {
49  
50      /**
51       * RESTful request.
52       */
53      private final transient Request request;
54  
55      /**
56       * Blob SHA hash.
57       */
58      private final transient String hash;
59  
60      /**
61       * Public ctor.
62       * @param req Request
63       * @param repo Repository
64       * @param sha Number of the get
65       */
66      RtBlob(
67          final Request req,
68          final Repo repo,
69          final String sha) {
70          final Coordinates coords = repo.coordinates();
71          this.request = req.uri()
72              .path("/repos")
73              .path(coords.user())
74              .path(coords.repo())
75              .path("/git")
76              .path("/blobs")
77              .path(sha)
78              .back();
79          this.hash = sha;
80      }
81  
82      @Override
83      public String sha() {
84          return this.hash;
85      }
86  
87      @Override
88      public JsonObject json() throws IOException {
89          return new RtJson(this.request).fetch();
90      }
91  }
