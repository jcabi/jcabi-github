5   package com.jcabi.github;
6   
7   import com.jcabi.aspects.Immutable;
8   import java.io.IOException;
9   import java.util.Map;
10  
11  /**
12   * GitHub hooks.
13   * @see <a href="https://developer.github.com/v3/repos/hooks/">Hooks API</a>
14   * @since 0.8
15   */
16  @Immutable
17  public interface Hooks {
18  
19      /**
20       * Owner of them.
21       * @return Repo
22       */
23      Repo repo();
24  
25      /**
26       * Iterate them all.
27       * @return Iterator of hooks
28       * @see <a href="https://developer.github.com/v3/repos/hooks/#list">List</a>
29       */
30      Iterable<Hook> iterate();
31  
32      /**
33       * Remove hook by ID.
34       * @param number ID of the label to remove
35       * @throws IOException If there is any I/O problem
36       * @see <a href="https://developer.github.com/v3/repos/hooks/#delete-a-hook">List</a>
37       */
38      void remove(int number) throws IOException;
39  
40      /**
41       * Get specific hook by number.
42       * @param number Hook number
43       * @return Hook
44       * @see <a href="https://developer.github.com/v3/repos/hooks/#get-single-hook">Get single hook</a>
45       */
46      Hook get(int number);
47  
48      /**
49       * Create new hook.
50       * @param name Hook name
51       * @param config Configuration for the hook
52       * @param events Events that trigger the hook
53       * @param active Actually trigger the hook when the events occur?
54       * @return Hook
55       * @throws IOException If there is any I/O problem
56       * @see <a href="https://developer.github.com/v3/repos/hooks/#create-a-hook">Create a hook</a>
57       * @checkstyle ParameterNumberCheck (2 lines)
58       */
59      Hook create(
60          String name, Map<String, String> config,
61          Iterable<Event> events, boolean active
62      ) throws IOException;
63  }
