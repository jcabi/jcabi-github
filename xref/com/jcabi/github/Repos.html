12  import jakarta.json.JsonObjectBuilder;
13  import jakarta.json.JsonValue;

15  import java.util.HashMap;

20  /**
21   * GitHub Repo API.
22   * @see <a href="https://developer.github.com/v3/repos/">Repos API</a>
23   * @since 0.5
24   */
25  @SuppressWarnings("PMD.TooManyMethods")
26  @Immutable
27  public interface Repos {
28  
29      /**
30       * Get its owner.
31       * @return GitHub
32       */
33      GitHub github();
34      //byte[]
35  
36      /**
37       * Create repository.
38       * @param settings Settings to use for creating the new repository
39       * @return Repository
40       * @throws IOException If there is any I/O problem
41       * @see <a href="https://developer.github.com/v3/repos/#create">Create Repository</a>
42       * @since 0.5
43       */
44      Repo create(Repos.RepoCreate settings)
45          throws IOException;
46  
47      /**
48       * Get repository by name.
49       * @param coords Repository name in "user/repo" format
50       * @return Repository
51       * @see <a href="https://developer.github.com/v3/repos/#get">Get Repository</a>
52       */
53      Repo get(Coordinates coords);
54  
55      /**
56       * Remove repository by name.
57       *
58       * <p>Note: Deleting a repository requires admin access.
59       * If OAuth is used, the delete_repo scope is required.
60       *
61       * @param coords Repository name in "user/repo" format
62       * @throws IOException If there is any I/O problem
63       * @see <a href="https://developer.github.com/v3/repos/#delete-a-repository">Delete a Repository</a>
64       */
65      void remove(Coordinates coords) throws IOException;
66  
67      /**
68       * Iterate all public repos, starting with the one you've seen already.
69       * @param identifier The integer ID of the last Repo that you’ve seen.
70       * @return Iterator of repo
71       * @see <a href="https://developer.github.com/v3/repos/#list-all-public-repositories">List all public repositories</a>
72       */
73      Iterable<Repo> iterate(
74          String identifier
75      );
76  
77      /**
78       * Check if a repository exists on GitHub.
79       * @param coords Coordinates of the repo.
80       * @return True if it exists, false otherwise.
81       * @throws IOException If something goes wrong.
82       */
83      boolean exists(Coordinates coords) throws IOException;
84  
85      /**
86       * Settings to use when creating a new GitHub repository.
87       * @see <a href="https://developer.github.com/v3/repos/#create">Create Repo API</a>
88       * @since 0.24
89       * @todo #1095:30m Add the ability to set the other parameters of
90       *  the repo creation API (has_issues, has_wiki, has_downloads,
91       *  team_id, gitignore_template, license_template).
92       */
93      @SuppressWarnings("PMD.TooManyMethods")
94      @ToString
95      @Loggable(Loggable.DEBUG)
96      @EqualsAndHashCode(of = {"repo", "priv", "descr", "home", "init"})
97      final class RepoCreate implements JsonReadable {
98          /**
99           * Name of the new repo.
100          */
101         private final transient String repo;
102 
103         /**
104          * Privateness of the new repo.
105          */
106         private final transient boolean priv;
107 
108         /**
109          * Description of the new repo.
110          */
111         private final transient String descr;
112 
113         /**
114          * Homepage of the new repo.
115          */
116         private final transient String home;
117 
118         /**
119          * Auto-init the new repo?
120          */
121         private final transient Optional<Boolean> init;
122 
123         /**
124          * Organization where the created repo belongs.
125          */
126         private final transient String organization;
127 
128         /**
129          * Other parameters which repo might have.
130          */
131         private final transient Map<String, JsonValue> other;
132 
133         /**
134          * Public ctor.
135          * @param nme Name of the new repository. Cannot be empty.
136          * @param prvt Will the new repo be private?
137          *  If not, then it will be public.
138          */
139         public RepoCreate(final String nme, final boolean prvt) {
140             this(nme, prvt, "", "", Optional.absent(), "");
141         }
142 
143         /**
144          * Private ctor.
145          * @param nme Name of the new repo. Cannot be empty.
146          * @param prvt Will the new repo be private?
147          *  If not, then it will be public.
148          * @param desc Description of the new repo
149          * @param page Homepage of the new repo
150          * @param auto Auto-init the new repo?
151          * @param org Organization to which this repo belongs.
152          *  When empty or null, the repo is created under the
153          *  authenticated user.
154          * @checkstyle ParameterNumberCheck (7 lines)
155          */
156         @SuppressWarnings("PMD.ConstructorOnlyInitializesOrCallOtherConstructors")
157         private RepoCreate(
158             final String nme,
159             final boolean prvt,
160             final String desc,
161             final String page,
162             final Optional<Boolean> auto,
163             final String org
164         ) {
165             if (nme.isEmpty()) {
166                 throw new IllegalArgumentException("Name cannot be empty!");
167             }
168             this.repo = nme;
169             this.priv = prvt;
170             this.descr = desc;
171             this.home = page;
172             this.init = auto;
173             this.organization = org;
174             this.other = new HashMap<>(0);
175         }
176 
177         /**
178          * Name of the new repo.
179          * @return Name
180          */
181         public String name() {
182             return this.repo;
183         }
184 
185         /**
186          * Will the new repo be private? If not, then it will be public.
187          * @return Is this repo private?
188          */
189         public boolean isPrivate() {
190             return this.priv;
191         }
192 
193         /**
194          * Description of the new repo.
195          * If it has no description, this is an empty string.
196          * @return Description
197          */
198         public String description() {
199             return this.descr;
200         }
201 
202         /**
203          * Homepage of the new repo.
204          * If it has no homepage, this is an empty string.
205          * @return Homepage
206          */
207         public String homepage() {
208             return this.home;
209         }
210 
211         /**
212          * Auto-init the new repo?
213          * If absent, the GitHub default will be used.
214          * @return Optional boolean
215          */
216         public Optional<Boolean> autoInit() {
217             return this.init;
218         }
219 
220         /**
221          * Name of the organization to which this repo belongs.
222          * @return String org name
223          */
224         public String organization() {
225             return this.organization;
226         }
227 
228         /**
229          * Returns a RepoCreate with the given name.
230          * The name cannot be empty.
231          * @param nme Name of the new repo
232          * @return RepoCreate
233          */
234         public Repos.RepoCreate withName(
235             final String nme
236         ) {
237             return new Repos.RepoCreate(
238                 nme,
239                 this.priv,
240                 this.descr,
241                 this.home,
242                 this.init,
243                 this.organization
244             );
245         }
246 
247         /**
248          * Returns a RepoCreate with the given privacy.
249          * @param privacy Privateness of the new repo
250          * @return RepoCreate
251          */
252         public Repos.RepoCreate withPrivacy(final boolean privacy) {
253             return new Repos.RepoCreate(
254                 this.repo,
255                 privacy,
256                 this.descr,

259                 this.organization
260             );
261         }

263         /**
264          * Returns a RepoCreate with the given description.
265          * @param desc Description
266          * @return RepoCreate
267          */
268         public Repos.RepoCreate withDescription(
269             final String desc
270         ) {
271             return new Repos.RepoCreate(

274                 desc,
275                 this.home,

277                 this.organization

279         }

285          */
286         public Repos.RepoCreate withHomepage(
287             final String page

289             return new Repos.RepoCreate(

300          * Returns a RepoCreate with the given auto-init enabledness.

304         public Repos.RepoCreate withAutoInit(final Optional<Boolean> auto) {

332          * Returns a RepoCreate with the given organization.

334          * @return RepoCreate

336         public Repos.RepoCreate withOrganization(final String org) {
337             return new Repos.RepoCreate(
338                 this.repo,

342                 this.init,
343                 org
344             );
345         }
346 
347         /**
348          * Returns a RepoCreate with the given json fields.
349          * @param key Json key
350          * @param value Json value
351          * @return The same RepoCreate.
352          * @todo #1660:30min Make 'with' method immutable.
353          *  Currently, the 'with' method mutates the 'other' field.
354          *  This is not ideal, as it makes the class mutable.
355          *  Make the 'with' method immutable and return a new
356          *  RepoCreate object with the new field.
357          */
358         public Repos.RepoCreate with(final String key, final JsonValue value) {
359             this.other.put(key, value);
360             return this;
361         }
362 
363         @Override
364         public JsonObject json() {
365             JsonObjectBuilder builder = Json.createObjectBuilder()
366                 .add("name", this.repo)
367                 .add("description", this.descr)
368                 .add("homepage", this.home)
369                 .add("private", this.priv);
370             if (this.init.isPresent()) {
371                 builder = builder.add("auto_init", this.init.get());
372             }
373             for (final Map.Entry<String, JsonValue> entry
374                 : this.other.entrySet()) {
375                 builder.add(entry.getKey(), entry.getValue());
376             }
377             return builder.build();
378         }
