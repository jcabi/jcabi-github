92      /**
93       * Smart user with extra features.
94       * @todo #1:30min Implement methods to retrieve all values provided
95       *  by Github for a single user, see:
96       *  http://developer.github.com/v3/users/#get-a-single-user
97       *  At the moment we implement just a few, but every data
98       *  items should have its own method. Of course, every new item should
99       *  be tested by a new unit test method.
100      * @see <a href="http://developer.github.com/v3/users/#get-a-single-user">Get a Single User</a>
101      */
102     @Immutable
103     @ToString
104     @Loggable(Loggable.DEBUG)
105     @EqualsAndHashCode(of = { "user", "jsn" })
106     final class Smart implements User {
107         /**
108          * Encapsulated user.
109          */
110         private final transient User user;
111         /**
112          * SmartJson object for convenient JSON parsing.
113          */
114         private final transient SmartJson jsn;
115 
116         /**
117          * Public ctor.
118          * @param usr User
119          */
120         public Smart(
121             @NotNull(message = "user can't be NULL") final User usr
122         ) {
123             this.user = usr;
124             this.jsn = new SmartJson(usr);
125         }
126 
127         /**
128          * Get his ID.
129          * @return Unique user ID
130          * @throws IOException If it fails
131          * @checkstyle MethodName (3 lines)
132          */
133         @SuppressWarnings("PMD.ShortMethodName")
134         public int id() throws IOException {
135             return this.user.json().getJsonNumber("id").intValue();
136         }

158         /**
159          * Get his name.
160          * @return User name
161          * @throws IOException If it fails
162          */
163         @NotNull(message = "Name is never NULL")
164         public String name() throws IOException {
165             final JsonObject json = this.json();
166             if (!json.containsKey("name")) {
167                 throw new IllegalStateException(
168                     String.format(
169                         // @checkstyle LineLength (1 line)
170                         "User %s doesn't have a name specified in his/her Github account; use #hasName() first.",
171                         this.login()
172                     )
173                 );
174             }
175             return json.getString("name");
176         }
177 
178         /**
179          * Check if user has name.
180          * @return True if user has name
181          * @throws IOException If it fails
182          */
183         public boolean hasName() throws IOException {
184             return this.json().containsKey("name");
185         }
