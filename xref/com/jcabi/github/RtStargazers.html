5   package com.jcabi.github;
6   
7   import com.jcabi.http.Request;
8   import com.jcabi.http.response.JsonResponse;
9   import jakarta.json.JsonReader;
10  import jakarta.json.JsonValue;
11  import java.io.IOException;
12  
13  /**
14   * GitHub stargazers.
15   * @since 1.7.1
16   */
17  public final class RtStargazers implements Stargazers {
18  
19      /**
20       * RESTful request.
21       */
22      private final transient Request request;
23  
24      /**
25       * Public ctor.
26       * @param entry Entry request.
27       */
28      RtStargazers(final Request entry) {
29          this.request = entry.uri()
30              .path("stargazers")
31              .back();
32      }
33  
34      @Override
35      public Iterable<JsonValue> iterable() throws IOException {
36          final Iterable<JsonValue> res;
37          try (
38              JsonReader json = new JsonResponse(
39                  this.request.method(Request.GET).fetch()
40              ).json()
41          ) {
42              res = json.readArray();
43          }
44          return res;
45      }
46  }
