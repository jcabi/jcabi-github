5   package com.jcabi.github;
6   
7   import com.jcabi.aspects.Immutable;
8   import java.io.IOException;
9   
10  /**
11   * GitHub forks.
12   * @see <a href="https://developer.github.com/v3/repos/forks/">Forks API</a>
13   * @since 0.8
14   */
15  @Immutable
16  public interface Forks {
17  
18      /**
19       * Owner of them.
20       * @return Repo
21       */
22      Repo repo();
23  
24      /**
25       * Iterate all forks.
26       * @param sort The sort order.
27       * @return All forks
28       * @see <a href="https://developer.github.com/v3/repos/forks/#list-forks">List forks</a>
29       */
30      Iterable<Fork> iterate(String sort);
31  
32      /**
33       * Create a fork for the authenticated user.
34       * @param organization The organization the repository will be forked into.
35       * @return The new fork
36       * @throws IOException  If there is any I/O problem
37       * @see <a href="https://developer.github.com/v3/repos/forks/#create-a-fork">Create a fork</a>
38       */
39      Fork create(String organization) throws IOException;
40  }
