5   package com.jcabi.github;
6   
7   import com.jcabi.aspects.Immutable;
8   import com.jcabi.aspects.Loggable;
9   import com.jcabi.http.Request;
10  import com.jcabi.http.response.JsonResponse;
11  import com.jcabi.http.response.RestResponse;
12  import jakarta.ws.rs.core.HttpHeaders;
13  import java.io.IOException;
14  import java.net.HttpURLConnection;
15  import java.net.URI;
16  import lombok.EqualsAndHashCode;
17  
18  /**
19   * GitHub release assets.
20   *
21   * @since 0.8
22   */
23  @Immutable
24  @Loggable(Loggable.DEBUG)
25  @EqualsAndHashCode(of = { "request", "owner" })
26  @SuppressWarnings("PMD.ConstructorOnlyInitializesOrCallOtherConstructors")
27  final class RtReleaseAssets implements ReleaseAssets {
28  
29      /**
30       * API entry point.
31       */
32      private final transient Request entry;
33  
34      /**
35       * RESTful request.
36       */
37      private final transient Request request;
38  
39      /**
40       * Owner of assets.
41       */
42      private final transient Release owner;
43  
44      /**
45       * Public ctor.
46       * @param req Request
47       * @param release Issue
48       */
49      RtReleaseAssets(
50          final Request req,
51          final Release release
52      ) {
53          this.entry = req;
54          final Coordinates coords = release.repo().coordinates();
55          // @checkstyle MultipleStringLiteralsCheck (7 lines)
56          this.request = this.entry.uri()
57              .path("/repos")
58              .path(coords.user())
59              .path(coords.repo())
60              .path("/releases")
61              .path(Integer.toString(release.number()))
62              .path("/assets")
63              .back();
64          this.owner = release;
65      }
66  
67      @Override
68      public Release release() {
69          return this.owner;
70      }
71  
72      @Override
73      public Iterable<ReleaseAsset> iterate() {
74          return new RtPagination<>(
75              this.request.uri().back()
76                  .method(Request.GET),
77              value -> this.get(
78                  //@checkstyle MultipleStringLiteralsCheck (1 line)
79                  value.getInt("id")
80              )
81          );
82      }
83  
84      @Override
85      public ReleaseAsset upload(
86          final byte[] content,
87          final String type,
88          final String name
89      ) throws IOException {
90          return this.get(
91              this.request.uri()
92                  .set(URI.create("https://uploads.github.com"))
93                  .path("/repos")
94                  .path(this.owner.repo().coordinates().user())
95                  .path(this.owner.repo().coordinates().repo())
96                  .path("/releases")
97                  .path(String.valueOf(this.owner.number()))
98                  .path("/assets")
99                  .queryParam("name", name)
100                 .back()
101                 .method(Request.POST)
102                 .reset(HttpHeaders.CONTENT_TYPE)
103                 .header(HttpHeaders.CONTENT_TYPE, type)
104                 .body().set(content).back()
105                 .fetch().as(RestResponse.class)
106                 .assertStatus(HttpURLConnection.HTTP_CREATED)
107                 .as(JsonResponse.class)
108                 .json().readObject().getInt("id")
109         );
110     }
111 
112     @Override
113     public ReleaseAsset get(final int number) {
114         return new RtReleaseAsset(this.entry, this.owner, number);
115     }
116 
117 }
