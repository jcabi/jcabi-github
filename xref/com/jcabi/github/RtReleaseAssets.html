30  package com.jcabi.github;
31  
32  import com.jcabi.aspects.Immutable;
33  import com.jcabi.aspects.Loggable;
34  import com.jcabi.http.Request;
35  import com.jcabi.http.response.JsonResponse;
36  import com.jcabi.http.response.RestResponse;
37  import java.io.IOException;
38  import java.net.HttpURLConnection;
39  import java.net.URI;
40  import javax.json.JsonObject;
41  import javax.ws.rs.core.HttpHeaders;
42  import lombok.EqualsAndHashCode;
43  
44  /**
45   * Github release assets.
46   *
47   * @author Carlos Miranda (miranda.cma@gmail.com)
48   * @version $Id: 0d5dcc9a247ddad4166980c89381e328caa4deae $
49   * @since 0.8
50   */
51  @Immutable
52  @Loggable(Loggable.DEBUG)
53  @EqualsAndHashCode(of = { "request", "owner" })
54  final class RtReleaseAssets implements ReleaseAssets {
55  
56      /**
57       * API entry point.
58       */
59      private final transient Request entry;
60  
61      /**
62       * RESTful request.
63       */
64      private final transient Request request;
65  
66      /**
67       * Owner of assets.
68       */
69      private final transient Release owner;
70  
71      /**
72       * Public ctor.
73       * @param req Request
74       * @param release Issue
75       */
76      RtReleaseAssets(
77          final Request req,
78          final Release release
79      ) {
80          this.entry = req;
81          final Coordinates coords = release.repo().coordinates();
82          // @checkstyle MultipleStringLiteralsCheck (7 lines)
83          this.request = this.entry.uri()
84              .path("/repos")
85              .path(coords.user())
86              .path(coords.repo())
87              .path("/releases")
88              .path(Integer.toString(release.number()))
89              .path("/assets")
90              .back();
91          this.owner = release;
92      }
93  
94      @Override
95      public Release release() {
96          return this.owner;
97      }
98  
99      @Override
100     public Iterable<ReleaseAsset> iterate() {
101         return new RtPagination<ReleaseAsset>(
102             this.request.uri().back()
103                 .method(Request.GET),
104             new RtValuePagination.Mapping<ReleaseAsset, JsonObject>() {
105                 @Override
106                 public ReleaseAsset map(final JsonObject value) {
107                     return RtReleaseAssets.this.get(
108                     //@checkstyle MultipleStringLiteralsCheck (1 line)
109                         value.getInt("id")
110                     );
111                 }
112             }
113         );
114     }
115 
116     @Override
117     public ReleaseAsset upload(
118         final byte[] content,
119         final String type,
120         final String name
121     ) throws IOException {
122         return this.get(
123             this.request.uri()
124                 .set(URI.create("https://uploads.github.com"))
125                 .path("/repos")
126                 .path(this.owner.repo().coordinates().user())
127                 .path(this.owner.repo().coordinates().repo())
128                 .path("/releases")
129                 .path(String.valueOf(this.owner.number()))
130                 .path("/assets")
131                 .queryParam("name", name)
132                 .back()
133                 .method(Request.POST)
134                 .reset(HttpHeaders.CONTENT_TYPE)
135                 .header(HttpHeaders.CONTENT_TYPE, type)
136                 .body().set(content).back()
137                 .fetch().as(RestResponse.class)
138                 .assertStatus(HttpURLConnection.HTTP_CREATED)
139                 .as(JsonResponse.class)
140                 .json().readObject().getInt("id")
141         );
142     }
143 
144     @Override
145     public ReleaseAsset get(final int number) {
146         return new RtReleaseAsset(this.entry, this.owner, number);
147     }
148 
149 }
