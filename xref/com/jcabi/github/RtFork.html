5   package com.jcabi.github;
6   
7   import com.jcabi.aspects.Immutable;
8   import com.jcabi.aspects.Loggable;
9   import com.jcabi.http.Request;
10  import jakarta.json.JsonObject;
11  import java.io.IOException;
12  import lombok.EqualsAndHashCode;
13  
14  /**
15   * GitHub fork.
16   *
17   * @since 0.8
18   */
19  @Immutable
20  @Loggable(Loggable.DEBUG)
21  @EqualsAndHashCode(of = {"request", "num" })
22  @SuppressWarnings("PMD.ConstructorOnlyInitializesOrCallOtherConstructors")
23  final class RtFork implements Fork {
24  
25      /**
26       * RESTful request.
27       */
28      private final transient Request request;
29  
30      /**
31       * Fork number.
32       */
33      private final transient int num;
34  
35      /**
36       * Public ctor.
37       * @param req Request
38       * @param repo Repository
39       * @param number Number of the get
40       */
41      RtFork(final Request req, final Repo repo, final int number) {
42          final Coordinates coords = repo.coordinates();
43          this.request = req.uri()
44              .path("/repos")
45              .path(coords.user())
46              .path(coords.repo())
47              .path("/forks")
48              .path(Integer.toString(number))
49              .back();
50          this.num = number;
51      }
52  
53      @Override
54      public int number() {
55          return this.num;
56      }
57  
58      @Override
59      public void patch(
60          final JsonObject json)
61          throws IOException {
62          new RtJson(this.request).patch(json);
63      }
64  
65      @Override
66      public JsonObject json() throws IOException {
67          return new RtJson(this.request).fetch();
68      }
69  }
