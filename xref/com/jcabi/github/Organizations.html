5   package com.jcabi.github;
6   
7   import com.jcabi.aspects.Immutable;
8   
9   /**
10   * GitHub organizations.
11   * @see <a href="https://developer.github.com/v3/orgs/">Organizations API</a>
12   * @since 0.24
13   */
14  @Immutable
15  public interface Organizations {
16      /**
17       * Get specific organization by name.
18       * @param login Login name of the organization.
19       * @return Organization
20       * @see <a href="https://developer.github.com/v3/orgs/#get-an-organization">Get a Single Organization</a>
21       */
22      Organization get(String login);
23  
24      /**
25       * Iterate over organizations of the currently logged-in user.
26       * @return Iterator of Organizations
27       * @see <a href="https://developer.github.com/v3/orgs/#list-your-organizations">List Your Organizations</a>
28       */
29      Iterable<Organization> iterate();
30  }
