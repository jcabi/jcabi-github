5   package com.jcabi.github;
6   
7   import java.util.Arrays;
8   
9   /**
10   * Runtime Reaction.
11   * @since 1.0
12   */
13  @SuppressWarnings("PMD.BeanMembersShouldSerialize")
14  public final class RtReaction implements Reaction {
15  
16      /**
17       * Origin Reaction.
18       */
19      private final Reaction origin;
20  
21      /**
22       * Constructor.
23       * @param reaction Decorated reaction.
24       */
25      public RtReaction(final Reaction reaction) {
26          this.origin = reaction;
27      }
28  
29      @Override
30      public String type() {
31          if (Arrays.asList(
32              Reaction.CONFUSED, Reaction.HEART, Reaction.HOORAY,
33              Reaction.LAUGH, Reaction.THUMBSDOWN, Reaction.THUMBSUP
34          ).contains(this.origin.type())
35          ) {
36              return this.origin.type();
37          }
38          throw new IllegalArgumentException(
39              String.format(
40                  "Invalid reaction: \"%s\"",
41                  this.origin.type()
42              )
43          );
44      }
45  }
