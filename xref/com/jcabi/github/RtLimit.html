5   package com.jcabi.github;
6   
7   import com.jcabi.aspects.Immutable;
8   import com.jcabi.aspects.Loggable;
9   import com.jcabi.http.Request;
10  import jakarta.json.JsonObject;
11  import java.io.IOException;
12  import lombok.EqualsAndHashCode;
13  
14  /**
15   * GitHub limit rate.
16   *
17   * @since 0.6
18   */
19  @Immutable
20  @Loggable(Loggable.DEBUG)
21  @EqualsAndHashCode(of = { "ghub", "entry", "res" })
22  final class RtLimit implements Limit {
23  
24      /**
25       * API entry point.
26       */
27      private final transient Request entry;
28  
29      /**
30       * GitHub.
31       */
32      private final transient GitHub ghub;
33  
34      /**
35       * Name of resource.
36       */
37      private final transient String res;
38  
39      /**
40       * Public ctor.
41       * @param github GitHub
42       * @param req Request
43       * @param name Name of resource
44       */
45      RtLimit(final GitHub github, final Request req, final String name) {
46          this.entry = req;
47          this.ghub = github;
48          this.res = name;
49      }
50  
51      @Override
52      public GitHub github() {
53          return this.ghub;
54      }
55  
56      @Override
57      public JsonObject json() throws IOException {
58          final JsonObject json = new RtJson(this.entry)
59              .fetch()
60              .getJsonObject("resources");
61          if (!json.containsKey(this.res)) {
62              throw new IllegalStateException(
63                  String.format(
64                      "'%s' is absent in JSON: %s", this.res, json
65                  )
66              );
67          }
68          return json.getJsonObject(this.res);
69      }
70  }
