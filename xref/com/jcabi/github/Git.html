30  package com.jcabi.github;
31  
32  import com.jcabi.aspects.Immutable;
33  import java.io.IOException;
34  import javax.validation.constraints.NotNull;
35  
36  /**
37   * Github Git.
38   *
39   * @author Carlos Miranda (miranda.cma@gmail.com)
40   * @version $Id$
41   * @since 0.8
42   */
43  @Immutable
44  public interface Git {
45  
46      /**
47       * Owner of it.
48       * @return Repo
49       */
50      @NotNull(message = "repository is never NULL")
51      Repo repo();
52  
53      /**
54       * Get its blobs.
55       * @return Blobs
56       * @see <a href="http://developer.github.com/v3/git/blobs/">Blobs API</a>
57       * @throws IOException If some io problem occurs
58       */
59      @NotNull(message = "Blobs is never NULL")
60      Blobs blobs() throws IOException;
61  
62      /**
63       * Get its commits.
64       * @return Commits
65       * @see <a href="http://developer.github.com/v3/git/commits/">Commits API</a>
66       * @todo #115:30min Implement the Git Data Commits API. We need to
67       *  implement the following:
68       *  1) Add the operations in http://developer.github.com/v3/git/commits/
69       *  to the Commits interface. Fetch operations should return Commit
70       *  instances.
71       *  2) A class RtCommits, which performs RESTful request operations
72       *  against the Commits API.
73       *  3) A class MkCommits, which mocks the Commits interface.
74       *  4) MkCommit, which mocks a single Commit object.
75       *  Don't forget to implement unit tests and integration tests.
76       */
77      @NotNull(message = "Commits is never NULL")
78      Commits commits();
79  
80      /**
81       * Get its references.
82       * @return References
83       * @see <a href="http://developer.github.com/v3/git/references/">References API</a>
84       */
85      @NotNull(message = "References is never NULL")
86      References references();
87  
88      /**
89       * Get its tags.
90       * @return Tags
91       * @see <a href="http://developer.github.com/v3/git/tags/">Tags API</a>
92       */
93      @NotNull(message = "Tags is never NULL")
94      Tags tags();
95  
96      /**
97       * Get its trees.
98       * @return Trees
99       * @see <a href="http://developer.github.com/v3/git/trees/">Trees API</a>
100      * @todo #115:1hr Implement the Git Data Trees API. We need to
101      *  implement the following:
102      *  1) Add the operations in http://developer.github.com/v3/git/trees/
103      *  to the Trees interface.
104      *  2) A class RtTrees, which performs RESTful request operations
105      *  against the Trees API.
106      *  3) A class MkTrees, which mocks the Trees interface.
107      *  4) An interface Tag, which represents a single Tag of the
108      *  Git Data API.
109      *  5) RtTrees, a Tag implementation which fetches and modifies
110      *  Tag information according to the Trees API, and
111      *  6) MkTag, which mocks a single Tag object.
112      *  Don't forget to implement unit tests and integration tests.
113      */
114     @NotNull(message = "Trees is never NULL")
115     Trees trees();
116 }
