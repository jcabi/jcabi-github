5   package com.jcabi.github;
6   
7   import com.jcabi.aspects.Immutable;
8   import java.io.IOException;
9   
10  /**
11   * GitHub Git.
12   * @since 0.8
13   */
14  @Immutable
15  public interface Git {
16  
17      /**
18       * Owner of it.
19       * @return Repo
20       */
21      Repo repo();
22  
23      /**
24       * Get its blobs.
25       * @return Blobs
26       * @throws IOException If some io problem occurs
27       * @see <a href="https://developer.github.com/v3/git/blobs/">Blobs API</a>
28       */
29      Blobs blobs() throws IOException;
30  
31      /**
32       * Get its commits.
33       * @return Commits
34       * @see <a href="https://developer.github.com/v3/git/commits/">Commits API</a>
35       */
36      Commits commits();
37  
38      /**
39       * Get its references.
40       * @return References
41       * @see <a href="https://developer.github.com/v3/git/references/">References API</a>
42       */
43      References references();
44  
45      /**
46       * Get its tags.
47       * @return Tags
48       * @see <a href="https://developer.github.com/v3/git/tags/">Tags API</a>
49       */
50      Tags tags();
51  
52      /**
53       * Get its trees.
54       * @return Trees
55       * @see <a href="https://developer.github.com/v3/git/trees/">Trees API</a>
56       */
57      Trees trees();
58  }
