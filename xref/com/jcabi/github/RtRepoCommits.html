30  package com.jcabi.github;
31  
32  import com.jcabi.aspects.Immutable;
33  import com.jcabi.aspects.Loggable;
34  import com.jcabi.http.Request;
35  import com.jcabi.http.RequestURI;
36  import com.jcabi.http.response.RestResponse;
37  import java.io.IOException;
38  import java.net.HttpURLConnection;
39  import java.util.Map;
40  import javax.json.JsonObject;
41  import javax.ws.rs.core.HttpHeaders;
42  import lombok.EqualsAndHashCode;
43  
44  /**
45   * Commits of a Github repository.
46   * @author Alexander Sinyagin (sinyagin.alexander@gmail.com)
47   * @version $Id: 193bccb764fd50b2d688459c0db507f2e97d6a39 $
48   * @checkstyle MultipleStringLiteralsCheck (200 lines)
49   */
50  @Immutable
51  @Loggable(Loggable.DEBUG)
52  @EqualsAndHashCode(of = { "request", "owner", "entry" })
53  final class RtRepoCommits implements RepoCommits {
54  
55      /**
56       * RESTful API entry point.
57       */
58      private final transient Request entry;
59  
60      /**
61       * RESTful request for the commits.
62       */
63      private final transient Request request;
64  
65      /**
66       * RESTful request for the comparison.
67       */
68      private final transient Request comp;
69  
70      /**
71       * Parent repository.
72       */
73      private final transient Repo owner;
74  
75      /**
76       * Public ctor.
77       * @param req Entry point of API
78       * @param repo Repository
79       */
80      RtRepoCommits(final Request req, final Repo repo) {
81          this.entry = req;
82          this.owner = repo;
83          final RequestURI rep = req.uri()
84              .path("/repos")
85              .path(repo.coordinates().user())
86              .path(repo.coordinates().repo());
87          this.request = rep
88              .path("/commits")
89              .back();
90          this.comp = rep
91              .path("/compare")
92              .back();
93      }
94  
95      @Override
96      public  Iterable<RepoCommit> iterate(
97          final Map<String, String> params
98      ) {
99          return new RtPagination<>(
100             this.request.uri().queryParams(params).back(),
101             value -> this.get(value.getString("sha"))
102         );
103     }
104 
105     @Override
106     public RepoCommit get(
107         final String sha
108     ) {
109         return new RtRepoCommit(this.entry, this.owner, sha);
110     }
111 
112     @Override
113     public CommitsComparison compare(
114         final String base,
115         final String head) {
116         return new RtCommitsComparison(this.entry, this.owner, base, head);
117     }
118 
119     @Override
120     public String diff(
121         final String base,
122         final String head)
123         throws IOException {
124         return this.comp.reset(HttpHeaders.ACCEPT)
125             .header(HttpHeaders.ACCEPT, "application/vnd.github.v3.diff")
126             .uri()
127             .path(String.format("%s...%s", base, head))
128             .back()
129             .fetch().as(RestResponse.class)
130             .assertStatus(HttpURLConnection.HTTP_OK)
131             .body();
132     }
133 
134     @Override
135     public String patch(
136         final String base,
137         final String head)
138         throws IOException {
139         return this.comp.reset(HttpHeaders.ACCEPT)
140             .header(HttpHeaders.ACCEPT, "application/vnd.github.v3.patch")
141             .uri()
142             .path(String.format("%s...%s", base, head))
143             .back()
144             .fetch().as(RestResponse.class)
145             .assertStatus(HttpURLConnection.HTTP_OK)
146             .body();
147     }
148 
149     @Override
150     public String toString() {
151         return this.request.uri().get().toString();
152     }
153 
154     @Override
155     public JsonObject json() throws IOException {
156         return new RtJson(this.request).fetch();
157     }
158 }
