5   package com.jcabi.github;
6   
7   import com.jcabi.aspects.Immutable;
8   
9   /**
10   * GitHub Notifications API.
11   * @see <a href="https://developer.github.com/v3/activity/notifications/">Notifications API</a>
12   * @since 0.15
13   */
14  @Immutable
15  public interface Notifications {
16      /**
17       * Iterate them all.
18       * @return Iterable of Notifications
19       * @see <a href="https://developer.github.com/v3/activity/notifications/#list-your-notifications-in-a-repository">List your notifications in a repository</a>
20       */
21      Iterable<Notification> iterate();
22  
23      /**
24       * Get a single notification.
25       * @param number Notification id
26       * @return Notification
27       * @see <a href="https://developer.github.com/v3/activity/notifications/#view-a-single-thread">View a single thread</a>
28       */
29      Notification get(int number);
30  
31      /**
32       * Marks all notifications on this repository as read.
33       * @see <a href="https://developer.github.com/v3/activity/notifications/#mark-notifications-as-read-in-a-repository">Mark notifications as read in a repository</a>
34       */
35      void markAsRead();
36  
37      /**
38       * Get thread data.
39       * @param number Thread ID.
40       * @return Data of the specified thread.
41       * @see <a href="https://developer.github.com/v3/activity/notifications/#view-a-single-thread">View a single thread</a>
42       */
43      GitHubThread thread(int number);
44  }
