5   package com.jcabi.github;
6   
7   import com.jcabi.aspects.Immutable;
8   import java.io.IOException;
9   
10  /**
11   * Public members of a GitHub organization.
12   *
13   * @see <a href="https://developer.github.com/v3/orgs/members/">Organization Members API</a>
14   * @since 0.24
15   */
16  @Immutable
17  public interface PublicMembers {
18      /**
19       * Organization of which these are public members.
20       * @return Organization
21       */
22      Organization org();
23  
24      /**
25       * Conceal a user's membership from public view.
26       * @param user User whose membership to conceal
27       * @throws IOException If an I/O problem occurs
28       * @see <a href="https://developer.github.com/v3/orgs/members/#conceal-a-users-membership">Conceal a user's membership</a>
29       */
30      void conceal(
31          User user
32      ) throws IOException;
33  
34      /**
35       * Make a user's membership publicly visible.
36       * @param user User whose membership to publicize
37       * @throws IOException If an I/O problem occurs
38       * @see <a href="https://developer.github.com/v3/orgs/members/#publicize-a-users-membership">Publicize a user's membership</a>
39       */
40      void publicize(
41          User user
42      ) throws IOException;
43  
44      /**
45       * Get all users who are public members of this organization.
46       * @return Members
47       * @see <a href="https://developer.github.com/v3/orgs/members/#public-members-list">Public members list</a>
48       */
49      Iterable<User> iterate();
50  
51      /**
52       * Check whether the user is a public member of this organization.
53       * @param user User to check public organization membership of
54       * @return Is the user a public member of this organization?
55       * @throws IOException If an I/O problem occurs
56       * @see <a href="https://developer.github.com/v3/orgs/members/#check-public-membership">Check public membership</a>
57       */
58      boolean contains(
59          User user
60      ) throws IOException;
61  }
