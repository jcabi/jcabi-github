30  package com.jcabi.github;
31  
32  import com.jcabi.aspects.Immutable;
33  import com.jcabi.aspects.Loggable;
34  import java.io.StringReader;
35  import javax.json.Json;
36  import javax.json.JsonObject;
37  import lombok.EqualsAndHashCode;
38  
39  /**
40   * Github commit status.
41   * @author Marcin Cylke (maracin.cylke+github@gmail.com)
42   * @author Chris Rebert (github@chrisrebert.com)
43   * @version $Id: 16a3b76096f6c801aeb4af282d9683d94ca343e9 $
44   * @since 0.23
45   */
46  @Immutable
47  @Loggable(Loggable.DEBUG)
48  @EqualsAndHashCode(of = { "cmmt", "jsn" })
49  public final class RtStatus implements Status {
50      /**
51       * Associated commit.
52       */
53      private final transient Commit cmmt;
54      /**
55       * Encapsulated status JSON.
56       */
57      private final transient String jsn;
58  
59      /**
60       * Public ctor.
61       * @param cmt Associated commit
62       * @param obj Status JSON object
63       */
64      public RtStatus(
65          final Commit cmt,
66          final JsonObject obj
67      ) {
68          this.cmmt = cmt;
69          this.jsn = obj.toString();
70      }
71  
72      @Override
73      public JsonObject json() {
74          return Json.createReader(new StringReader(this.jsn)).readObject();
75      }
76  
77      @Override
78      public int identifier() {
79          return this.json().getInt("id");
80      }
81  
82      @Override
83      public String url() {
84          return this.json().getString("url");
85      }
86  
87      @Override
88      public Commit commit() {
89          return this.cmmt;
90      }
91  }
