5   package com.jcabi.github;
6   
7   import com.jcabi.aspects.Immutable;
8   import com.jcabi.aspects.Loggable;
9   import java.io.IOException;
10  import lombok.EqualsAndHashCode;
11  
12  /**
13   * Check if a GitHub JSON object actually exists on the server.
14   * @since 0.38
15   */
16  @Immutable
17  @Loggable(Loggable.DEBUG)
18  @EqualsAndHashCode(of = "readable")
19  final class Existence {
20  
21      /**
22       * Checked object.
23       */
24      private final transient JsonReadable readable;
25  
26      /**
27       * Ctor.
28       * @param rdbl Checked object.
29       */
30      Existence(final JsonReadable rdbl) {
31          this.readable = rdbl;
32      }
33  
34      /**
35       * Check existence.
36       * @return True if exists, false otherwise.
37       * @throws IOException If there is a networking problem.
38       */
39      public boolean check() throws IOException {
40          boolean exists = true;
41          try {
42              this.readable.json();
43          } catch (final AssertionError | IndexOutOfBoundsException ex) {
44              exists = false;
45          }
46          return exists;
47      }
48  
49  }
