5   package com.jcabi.github;
6   
7   import com.jcabi.aspects.Immutable;
8   import com.jcabi.http.Request;
9   import org.apache.commons.lang3.NotImplementedException;
10  
11  /**
12   * GitHub Notifications.
13   *
14   * @see <a href="https://developer.github.com/v3/activity/notifications/">Notifications API</a>
15   * @since 0.15
16   * @todo #913:30min Implement markAsRead(), thread(final int number) operations
17   *  in RtNotifications. Don't forget about unit tests.
18   */
19  @Immutable
20  final class RtNotifications implements Notifications {
21      /**
22       * RESTful request.
23       */
24      private final transient Request request;
25  
26      /**
27       * Ctor.
28       * @param req The request for this notifications.
29       */
30      RtNotifications(final Request req) {
31          this.request = req;
32      }
33  
34      @Override
35      public Iterable<Notification> iterate() {
36          return new RtPagination<>(
37              this.request.uri()
38                  .queryParam("all", "true")
39                  .queryParam("since", "1970-01-01T00:00:00Z")
40                  .back(),
41              json -> new RtNotification(
42                  Long.valueOf(json.getString("id"))
43              )
44          );
45      }
46  
47      @Override
48      public Notification get(final int number) {
49          return new RtNotification(number);
50      }
51  
52      @Override
53      public void markAsRead() {
54          throw new NotImplementedException("RtNotifications#markAsRead");
55      }
56  
57      @Override
58      public GitHubThread thread(final int number) {
59          throw new NotImplementedException("RtNotifications#thread");
60      }
61  }
