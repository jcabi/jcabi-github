30  package com.jcabi.github;
31  
32  import com.jcabi.aspects.Immutable;
33  import com.jcabi.http.Request;
34  import org.apache.commons.lang3.NotImplementedException;
35  
36  /**
37   * Github Notifications.
38   *
39   * @author Giang Le (lthuangiang@gmail.com)
40   * @author Paul Polishchuk (ppol@ua.fm)
41   * @version $Id: c516f1e24cd540c9c470da718ed8d5ae20822c62 $
42   * @since 0.15
43   * @see <a href="https://developer.github.com/v3/activity/notifications/">Notifications API</a>
44   * @todo #913:30min Implement markAsRead(), thread(final int number) operations
45   *  in RtNotifications. Don't forget about unit tests.
46   */
47  @Immutable
48  final class RtNotifications implements Notifications {
49      /**
50       * RESTful request.
51       */
52      private final transient Request request;
53  
54      /**
55       * Ctor.
56       * @param req The request for this notifications.
57       */
58      RtNotifications(final Request req) {
59          this.request = req;
60      }
61  
62      @Override
63      public Iterable<Notification> iterate() {
64          return new RtPagination<>(
65              this.request.uri()
66                  .queryParam("all", "true")
67                  .queryParam("since", "1970-01-01T00:00:00Z")
68                  .back(),
69              json -> new RtNotification(
70                  Long.valueOf(json.getString("id"))
71              )
72          );
73      }
74  
75      @Override
76      public Notification get(final int number) {
77          return new RtNotification(number);
78      }
79  
80      @Override
81      public void markAsRead() {
82          throw new NotImplementedException("RtNotifications#markAsRead");
83      }
84  
85      @Override
86      public GitHubThread thread(final int number) {
87          throw new NotImplementedException("RtNotifications#thread");
88      }
89  }
