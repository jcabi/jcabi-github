5   package com.jcabi.github;
6   
7   import com.jcabi.aspects.Immutable;
8   import java.io.IOException;
9   
10  /**
11   * Gist Comments.
12   *
13   * <p>Use this class to get access to all comments in a gist, for example:
14   *
15   * <pre> gist = // ... get it somewhere
16   * GistComments comments = gist.comments();
17   * GistComment comment = comments.post("Hi, how are you?");</pre>
18   *
19   * @see <a href="https://developer.github.com/v3/gists/comments/">Gist Comments API</a>
20   * @since 0.8
21   */
22  @Immutable
23  public interface GistComments {
24      /**
25       * The gist we're in.
26       * @return Issue
27       */
28      Gist gist();
29  
30      /**
31       * Get comment by number.
32       * @param number Comment number
33       * @return Comment
34       * @see <a href="https://developer.github.com/v3/gists/comments/#get-a-single-comment">Get a Single Comment</a>
35       */
36      GistComment get(int number);
37  
38      /**
39       * Iterate them all.
40       * @return All comments
41       * @see <a href="https://developer.github.com/v3/gists/comments/#list-comments-on-a-gist">List Comments on an Gist</a>
42       */
43      Iterable<GistComment> iterate();
44  
45      /**
46       * Post new comment.
47       * @param text Text of comment to post in Markdown format
48       * @return Comment
49       * @throws IOException If there is any I/O problem
50       * @see <a href="https://developer.github.com/v3/gists/comments/#create-a-comment">Create a Comment</a>
51       */
52      GistComment post(String text)
53          throws IOException;
54  }
