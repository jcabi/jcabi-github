30  package com.jcabi.github;
31  
32  import com.jcabi.aspects.Immutable;
33  import java.io.IOException;
34  
35  /**
36   * Gist Comments.
37   *
38   * <p>Use this class to get access to all comments in a gist, for example:
39   *
40   * <pre> gist = // ... get it somewhere
41   * GistComments comments = gist.comments();
42   * GistComment comment = comments.post("Hi, how are you?");</pre>
43   *
44   * @author Giang Le (giang@vn-smartsolutions.com)
45   * @version $Id: cde418830e0581fe8a18b886db1c0ad5bdebeeca $
46   * @since 0.8
47   * @see <a href="http://developer.github.com/v3/gists/comments/">Gist Comments API</a>
48   */
49  @Immutable
50  public interface GistComments {
51      /**
52       * The gist we're in.
53       * @return Issue
54       */
55      Gist gist();
56  
57      /**
58       * Get comment by number.
59       * @param number Comment number
60       * @return Comment
61       * @see <a href="http://developer.github.com/v3/gists/comments/#get-a-single-comment">Get a Single Comment</a>
62       */
63      GistComment get(int number);
64  
65      /**
66       * Iterate them all.
67       * @return All comments
68       * @see <a href="http://developer.github.com/v3/gists/comments/#list-comments-on-a-gist">List Comments on an Gist</a>
69       */
70      Iterable<GistComment> iterate();
71  
72      /**
73       * Post new comment.
74       * @param text Text of comment to post in Markdown format
75       * @return Comment
76       * @throws IOException If there is any I/O problem
77       * @see <a href="http://developer.github.com/v3/gists/comments/#create-a-comment">Create a Comment</a>
78       */
79      GistComment post(String text)
80          throws IOException;
81  }
