5   package com.jcabi.github;
6   
7   import com.jcabi.aspects.Immutable;
8   import com.jcabi.aspects.Loggable;
9   import com.jcabi.http.Request;
10  import lombok.EqualsAndHashCode;
11  
12  /**
13   * GitHub organizations.
14   * @see <a href="https://developer.github.com/v3/orgs/">Organizations API</a>
15   * @since 0.24
16   */
17  @Immutable
18  @Loggable(Loggable.DEBUG)
19  @EqualsAndHashCode(of = { "entry", "ghub", "request" })
20  final class RtOrganizations implements Organizations {
21      /**
22       * API entry point.
23       */
24      private final transient Request entry;
25  
26      /**
27       * RESTful request.
28       */
29      private final transient Request request;
30  
31      /**
32       * GitHub.
33       */
34      private final transient GitHub ghub;
35  
36      /**
37       * Public ctor.
38       * @param github GitHub
39       * @param req Request
40       */
41      RtOrganizations(final GitHub github, final Request req) {
42          this.entry = req;
43          this.request = this.entry.uri().path("/user").path("/orgs").back();
44          this.ghub = github;
45      }
46  
47      @Override
48      public Organization get(
49          final String login
50      ) {
51          return new RtOrganization(this.ghub, this.entry, login);
52      }
53  
54      @Override
55      public Iterable<Organization> iterate() {
56          return new RtPagination<>(
57              this.request,
58              object -> this.get(object.getString("login"))
59          );
60      }
61  }
