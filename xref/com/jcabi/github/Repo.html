15  /**
16   * GitHub repository.
17   * @since 0.1
18   * @checkstyle MultipleStringLiterals (500 lines)
19   * @checkstyle ClassFanOutComplexityCheck (500 lines)
20   */
21  @Immutable
22  @SuppressWarnings({"PMD.TooManyMethods", "PMD.ExcessivePublicCount", "PMD.CouplingBetweenObjects"})
23  public interface Repo extends JsonReadable, JsonPatchable, Comparable<Repo> {
24  
25      /**
26       * Get its owner.
27       * @return GitHub
28       */
29      GitHub github();
30  
31      /**
32       * Get its coordinates.
33       * @return Coordinates
34       */
35      Coordinates coordinates();
36  
37      /**
38       * Iterate issues.
39       * @return Issues
40       */
41      Issues issues();
42  
43      /**
44       * Iterate milestones.
45       * @return Milestones
46       * @since 0.7
47       */
48      Milestones milestones();
49  
50      /**
51       * Pull requests.
52       * @return Pulls
53       */
54      Pulls pulls();
55  
56      /**
57       * Hooks.
58       * @return Hooks
59       * @since 0.8
60       */
61      Hooks hooks();
62  
63      /**
64       * Get all issue events for the repository.
65       * @return Issue events
66       * @see <a href="https://developer.github.com/v3/issues/events/#list-events-for-a-repository">List Events for a Repository</a>
67       */
68      IssueEvents issueEvents();
69  
70      /**
71       * Get all labels of the repo.
72       * @return Labels
73       * @see <a href="https://developer.github.com/v3/issues/labels/">Labels API</a>
74       */
75      Labels labels();
76  
77      /**
78       * Get all available assignees to which issues may be assigned.
79       * @return Assignees
80       * @see <a href="https://developer.github.com/v3/issues/assignees/">Assignees API</a>
81       */
82      Assignees assignees();
83  
84      /**
85       * Get all releases of the repo.
86       * @return Releases
87       * @see <a href="https://developer.github.com/v3/repos/releases/">Releases API</a>
88       */
89      Releases releases();
90  
91      /**
92       * Get all deploy keys of the repo.
93       * @return DeployKeys
94       * @see <a href="https://developer.github.com/v3/repos/keys/">Deploy Keys API</a>
95       */
96      DeployKeys keys();
97  
98      /**
99       * Get all forks of the repo.
100      * @return Forks
101      * @see <a href="https://developer.github.com/v3/repos/forks/">Forks API</a>
102      */
103     Forks forks();
104 
105     /**
106      * Get repository's commits.
107      * @return Commits
108      * @see <a href="https://developer.github.com/v3/repos/commits/">Commits API</a>
109      */
110     RepoCommits commits();
111 
112     /**
113      * Get repository's branches.
114      * @return Branches
115      * @see <a href="https://developer.github.com/v3/repos/#list-branches">List Branches API</a>
116      */
117     Branches branches();
118 
119     /**
120      * Get all contents of the repo.
121      * @return Contents
122      * @see <a href="https://developer.github.com/v3/repos/contents/">Contents API</a>
123      */
124     Contents contents();
125 
126     /**
127      * Gel all collaborators.
128      * @return Collaborators
129      * @see <a href="https://developer.github.com/v3/repos/collaborators/">Collaborators API</a>
130      */
131     Collaborators collaborators();
132 
133     /**
134      * Get the Git API entry point.
135      * @return Collaborators
136      * @see <a href="https://developer.github.com/v3/git/">Git Data API</a>
137      */
138     Git git();
139 
140     /**
141      * Get Starring API.
142      * @return Stars
143      * @see <a href="https://developer.github.com/v3/activity/starring/">Starring API</a>
144      * @since 0.15
145      */
146     Stars stars();
147 
148     /**
149      * Get Notifications API.
150      * @return Stars
151      * @see <a href="https://developer.github.com/v3/activity/notifications/">Notifications API</a>
152      * @since 0.15
153      */
154     Notifications notifications();
155 
156     /**
157      * Get languages for the specified repository.
158      * @return Languages
159      * @throws IOException If there is any I/O problem
160      * @see <a href="https://developer.github.com/v3/repos/#list-languages">List languages</a>
161      * @since 0.15
162      */
163     Iterable<Language> languages() throws IOException;
164 
165     /**
166      * Get default branch.
167      *
168      * @return Default branch.
169      * @throws IOException If there is any I/O problem.
170      */
171     Branch defaultBranch() throws IOException;
172 
173     /**
174      * Lists the people that have starred the repository.
175      * @return Lists the people that have starred the repository.
176      */
177     Stargazers stargazers();

179     /**
180      * Smart Repo with extra features.
181      * @since 0.1
182      */
183     @Immutable
184     @ToString
185     @Loggable(Loggable.DEBUG)
186     @EqualsAndHashCode(of = {"repo", "jsn"})
187     final class Smart implements Repo {
188         /**
189          * Encapsulated Repo.
190          */
191         private final transient Repo repo;
192 
193         /**
194          * SmartJson object for convenient JSON parsing.
195          */
196         private final transient SmartJson jsn;
197 
198         /**
199          * Public ctor.
200          * @param rep Repo
201          */
202         public Smart(
203             final Repo rep
204         ) {
205             this.repo = rep;
206             this.jsn = new SmartJson(rep);
207         }
208 
209         /**
210          * Does this Repo actually exist in GitHub?
211          * @return True if it exists, false otherwise.
212          * @throws IOException If there is any I/O problem.
213          */
214         public boolean exists() throws IOException {
215             return new Existence(this.repo).check();
216         }
217 
218         /**
219          * Does it have a description.
220          * @return TRUE if description is present
221          * @throws IOException If there is any I/O problem
222          */
223         public boolean hasDescription() throws IOException {
224             return this.jsn.hasNotNull("description");
225         }
226 
227         /**
228          * Get its description.
229          * @return Description
230          * @throws IOException If there is any I/O problem
231          */
232         public String description() throws IOException {
233             return this.jsn.text("description");
234         }
235 
236         /**
237          * Is it private?.
238          * @return TRUE if it's private
239          * @throws IOException If there is any I/O problem
240          */
241         public boolean isPrivate() throws IOException {
242             return Boolean.parseBoolean(
243                 this.json()
244                     .getOrDefault("private", JsonValue.FALSE)
245                     .toString().replace("\"", "")
246             );
247         }
248 
249         @Override
250         public GitHub github() {
251             return this.repo.github();
252         }
253 
254         @Override
255         public Coordinates coordinates() {
256             return this.repo.coordinates();
257         }
258 
259         @Override
260         public Issues issues() {
261             return this.repo.issues();
262         }
263 
264         @Override
265         public Milestones milestones() {
266             return this.repo.milestones();
267         }
268 
269         @Override
270         public Pulls pulls() {
271             return this.repo.pulls();
272         }
273 
274         @Override
275         public Hooks hooks() {
276             return this.repo.hooks();
277         }
278 
279         @Override
280         public IssueEvents issueEvents() {
281             return this.repo.issueEvents();
282         }
283 
284         @Override
285         public Labels labels() {
286             return this.repo.labels();
287         }
288 
289         @Override
290         public Assignees assignees() {
291             return this.repo.assignees();
292         }
293 
294         @Override
295         public Releases releases() {
296             return this.repo.releases();
297         }
298 
299         @Override
300         public DeployKeys keys() {
301             return this.repo.keys();
302         }
303 
304         @Override
305         public Forks forks() {
306             return this.repo.forks();
307         }
308

310         public Contents contents() {
311             return this.repo.contents();
312         }
313 
314         @Override
315         public Collaborators collaborators() {
316             return this.repo.collaborators();
317         }

320         public Git git() {
321             return this.repo.git();
322         }

324         @Override
325         public Stars stars() {
326             return this.repo.stars();
327         }
328 
329         @Override
330         public Notifications notifications() {
331             return this.repo.notifications();
332         }
333 
334         @Override
335         public Iterable<Language> languages() throws IOException {
336             return this.repo.languages();
337         }
338 
339         @Override
340         public Branch defaultBranch() throws IOException {
341             return this.repo.defaultBranch();
342         }
343 
344         @Override
345         public Stargazers stargazers() {
346             throw new UnsupportedOperationException(
347                 "stargazers() not yet implemented"
348             );
349         }
350

352         public void patch(
353             final JsonObject json
354         ) throws IOException {
355             this.repo.patch(json);
356         }
357 
358         @Override
359         public RepoCommits commits() {
360             return this.repo.commits();
361         }
362 
363         @Override
364         public Branches branches() {
365             return this.repo.branches();
366         }
367 
368         @Override
369         public JsonObject json() throws IOException {
370             return this.repo.json();
371         }
372

374         public int compareTo(final Repo repos) {
375             return this.repo.compareTo(repos);
376         }
377     }
