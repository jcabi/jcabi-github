5   package com.jcabi.github.mock;
6   
7   import com.jcabi.aspects.Immutable;
8   import com.jcabi.github.Notification;
9   import com.jcabi.xml.XML;
10  
11  /**
12   * Mock for GitHub Notification.
13   * @see <a href="https://developer.github.com/v3/activity/notifications/">Notifications API</a>
14   * @since 0.25
15   */
16  @Immutable
17  final class MkNotification implements Notification {
18  
19      /**
20       * XML data for this mock notification.
21       */
22      private final transient XML data;
23  
24      /**
25       * Public ctor.
26       * @param xml XML holding the data for this notification.
27       */
28      MkNotification(final XML xml) {
29          this.data = xml;
30      }
31  
32      @Override
33      public long number() {
34          return Long.valueOf(this.data.xpath("//id/text()").get(0));
35      }
36  }
