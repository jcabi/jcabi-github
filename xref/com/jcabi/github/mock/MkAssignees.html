5   package com.jcabi.github.mock;
6   
7   import com.jcabi.aspects.Immutable;
8   import com.jcabi.github.Assignees;
9   import com.jcabi.github.Coordinates;
10  import com.jcabi.github.User;
11  import java.io.IOException;
12  import java.util.HashSet;
13  import java.util.List;
14  import java.util.Set;
15  
16  /**
17   * Mock for GitHub Assignees.
18   *
19   * @since 0.7
20   */
21  @Immutable
22  final class MkAssignees implements Assignees {
23  
24      /**
25       * Storage.
26       */
27      private final transient MkStorage storage;
28  
29      /**
30       * Login of the user logged in.
31       */
32      private final transient String self;
33  
34      /**
35       * Repo name.
36       */
37      private final transient Coordinates coords;
38  
39      /**
40       * Public ctor.
41       * @param stg Storage
42       * @param login User to login
43       * @param rep Repo
44       */
45      MkAssignees(
46          final MkStorage stg,
47          final String login,
48          final Coordinates rep
49      ) {
50          this.storage = stg;
51          this.self = login;
52          this.coords = rep;
53      }
54  
55      @Override
56      public Iterable<User> iterate() {
57          final Set<User> assignees = new HashSet<>();
58          assignees.add(new MkUser(this.storage, this.self));
59          final Iterable<User> collabs = new MkIterable<>(
60              this.storage,
61              this.xpath().concat("/user"),
62              xml -> new MkUser(
63                  this.storage,
64                  xml.xpath("login/text()").get(0)
65              )
66          );
67          for (final User collab : collabs) {
68              assignees.add(collab);
69          }
70          return assignees;
71      }
72  
73      @Override
74      public boolean check(
75          final String login
76      ) {
77          try {
78              final List<String> xpath = this.storage.xml().xpath(
79                  this.xpath().concat("/user/login/text()")
80              );
81              return this.self.equalsIgnoreCase(login) || !xpath.isEmpty()
82                  && login.equalsIgnoreCase(xpath.get(0));
83          } catch (final IOException ex) {
84              throw new IllegalStateException(ex);
85          }
86      }
87  
88      /**
89       * XPath of the Collaborators element in the XML tree.
90       * @return XPath
91       */
92      private String xpath() {
93          return String.format(
94              "/github/repos/repo[@coords='%s']/collaborators",
95              this.coords
96          );
97      }
98  }
