39  /**
40   * Mock GitHub repo.
41   * @since 0.5
42   * @todo #1061 Fix code to avoid CouplingBetweenObjects
43   * @checkstyle ClassDataAbstractionCouplingCheck (500 lines)
44   * @checkstyle ClassFanOutComplexity (500 lines)
45   */
46  @Immutable
47  @Loggable(Loggable.DEBUG)
48  @ToString
49  @SuppressWarnings(
50      {
51          "PMD.TooManyMethods",
52          "PMD.ExcessiveImports",
53          "PMD.CouplingBetweenObjects"
54      }
55  )
56  final class MkRepo implements Repo {
57  
58      /**
59       * Storage.
60       */
61      private final transient MkStorage storage;
62  
63      /**
64       * Login of the user logged in.
65       */
66      private final transient String self;
67  
68      /**
69       * Repo coordinates.
70       */
71      private final transient Coordinates coords;
72  
73      /**
74       * Public ctor.
75       * @param stg Storage
76       * @param login User to login
77       * @param repo Repo name
78       */
79      MkRepo(
80          final MkStorage stg,
81          final String login,
82          final Coordinates repo
83      ) {
84          this.storage = stg;
85          this.self = login;
86          this.coords = repo;
87      }
88  
89      @Override
90      public GitHub github() {
91          return new MkGitHub(this.storage, this.self);
92      }
93  
94      @Override
95      public Coordinates coordinates() {
96          return this.coords;
97      }
98

106     }

298     @Override
299     public int compareTo(final Repo repo) {
300         return this.coords.compareTo(repo.coordinates());
301     }
302 
303     @Override
304     public boolean equals(final Object obj) {
305         final boolean result;
306         if (this == obj) {
307             result = true;
308         } else if (obj == null || this.getClass() != obj.getClass()) {
309             result = false;
310         } else {
311             final MkRepo other = (MkRepo) obj;
312             result = this.storage.equals(other.storage)
313                 && this.self.equals(other.self)
314                 && this.coords.equals(other.coords);
315         }
316         return result;
317     }
318 
319     @Override
320     public int hashCode() {
321         int result = this.storage.hashCode();
322         result = 31 * result + this.self.hashCode();
323         result = 31 * result + this.coords.hashCode();
324         return result;
325     }

331     private String xpath() {
332         return String.format(
333             "/github/repos/repo[@coords='%s']",
334             this.coords
335         );
336     }
