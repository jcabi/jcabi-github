30  package com.jcabi.github.mock;
31  
32  import com.jcabi.aspects.Immutable;
33  import com.jcabi.aspects.Loggable;
34  import com.jcabi.github.Coordinates;
35  import com.jcabi.github.Event;
36  import com.jcabi.github.Repo;
37  import java.io.IOException;
38  import javax.json.Json;
39  import javax.json.JsonObject;
40  import javax.json.JsonObjectBuilder;
41  import javax.json.JsonValue;
42  import lombok.EqualsAndHashCode;
43  import lombok.ToString;
44  
45  /**
46   * Mock Github event.
47   *
48   * @author Yegor Bugayenko (yegor256@gmail.com)
49   * @version $Id: 4cf881389239632bb54df3f072e69b22f6a3af0d $
50   * @since 0.6.1
51   */
52  @Immutable
53  @Loggable(Loggable.DEBUG)
54  @ToString
55  @EqualsAndHashCode(of = { "storage", "self", "coords", "num" })
56  final class MkEvent implements Event {
57      /**
58       * Created at.
59       */
60      private static final String CREATED_AT = "created_at";
61      /**
62       * Event.
63       */
64      private static final String EVENT = "event";
65      /**
66       * Login.
67       */
68      private static final String LOGIN = "login";
69  
70      /**
71       * Storage.
72       */
73      private final transient MkStorage storage;
74  
75      /**
76       * Login of the user logged in.
77       */
78      private final transient String self;
79  
80      /**
81       * Repo name.
82       */
83      private final transient Coordinates coords;
84  
85      /**
86       * ID number of event.
87       */
88      private final transient int num;
89  
90      /**
91       * Public ctor.
92       * @param stg Storage
93       * @param login User to login
94       * @param rep Repo
95       * @param nmbr Number
96       * @checkstyle ParameterNumber (5 lines)
97       */
98      MkEvent(
99          final MkStorage stg,
100         final String login,
101         final Coordinates rep,
102         final int nmbr
103     ) {
104         this.storage = stg;
105         this.self = login;
106         this.coords = rep;
107         this.num = nmbr;
108     }
109 
110     @Override
111     public Repo repo() {
112         return new MkRepo(this.storage, this.self, this.coords);
113     }
114 
115     @Override
116     public int number() {
117         return this.num;
118     }
119 
120     @Override
121     public int compareTo(
122         final Event event
123     ) {
124         throw new UnsupportedOperationException("#compareTo()");
125     }
126 
127     /**
128      * Describes the event in a JSON object.
129      * @return JSON object
130      * @throws IOException If there is any I/O problem
131      * @todo #1063:30min When the event has a label, retrieve and include the
132      *  label's color too. MkIssueEvents.create() will also need to be
133      *  updated accordingly.
134      */
135     @Override
136     public JsonObject json() throws IOException {
137         final JsonObject obj = new JsonNode(
138             this.storage.xml().nodes(this.xpath()).get(0)
139         ).json();
140         JsonObjectBuilder builder = Json.createObjectBuilder()
141             .add("id", this.num)
142             .add(
143                 "url",
144                 String.format(
145                     // @checkstyle LineLength (1 line)
146                     "https://api.jcabi-github.invalid/repos/%s/issues/events/%s",
147                     this.coords,
148                     this.num
149                 )
150             )
151             .add("commit_id", JsonValue.NULL)
152             // @checkstyle MultipleStringLiteralsCheck (1 line)
153             .add(MkEvent.EVENT, obj.getString(MkEvent.EVENT))
154             .add(
155                 "actor",
156                 Json.createObjectBuilder()
157                     // @checkstyle MultipleStringLiteralsCheck (1 line)
158                     .add(MkEvent.LOGIN, obj.getString(MkEvent.LOGIN))
159                     .build()
160             )
161             // @checkstyle MultipleStringLiteralsCheck (1 line)
162             .add(MkEvent.CREATED_AT, obj.getString(MkEvent.CREATED_AT));
163         final String label = "label";
164         if (obj.containsKey(label)) {
165             builder = builder.add(
166                 label,
167                 Json.createObjectBuilder()
168                     .add("name", obj.getString(label))
169                     .build()
170             );
171         }
172         return builder.build();
173     }
174 
175     /**
176      * XPath of this element in XML tree.
177      * @return XPath
178      */
179     private String xpath() {
180         return String.format(
181             // @checkstyle LineLength (1 line)
182             "/github/repos/repo[@coords='%s']/issue-events/issue-event[number='%d']",
183             this.coords, this.num
184         );
185     }
186 }
