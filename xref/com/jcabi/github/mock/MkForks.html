30  package com.jcabi.github.mock;
31  
32  import com.jcabi.aspects.Immutable;
33  import com.jcabi.github.Coordinates;
34  import com.jcabi.github.Fork;
35  import com.jcabi.github.Forks;
36  import com.jcabi.github.Repo;
37  import com.jcabi.log.Logger;
38  import java.io.IOException;
39  import lombok.EqualsAndHashCode;
40  import org.xembly.Directives;
41  
42  /**
43   * Mock Github forks.
44   *
45   * @author Carlos Miranda (miranda.cma@gmail.com)
46   * @version $Id: 34ef1cafc39f55f0a6ea0d1c064de01d19eeaed6 $
47   */
48  @Immutable
49  @EqualsAndHashCode(of = { "storage", "self", "coords" })
50  final class MkForks implements Forks {
51  
52      /**
53       * Storage.
54       */
55      private final transient MkStorage storage;
56  
57      /**
58       * Login of the user logged in.
59       */
60      private final transient String self;
61  
62      /**
63       * Repo name.
64       */
65      private final transient Coordinates coords;
66  
67      /**
68       * Public ctor.
69       * @param stg Storage
70       * @param login User to login
71       * @param rep Repo
72       * @throws IOException If there is any I/O problem
73       */
74      public MkForks(
75          final MkStorage stg,
76          final String login,
77          final Coordinates rep
78      ) throws IOException {
79          this.storage = stg;
80          this.self = login;
81          this.coords = rep;
82          this.storage.apply(
83              new Directives().xpath(
84                  String.format(
85                      "/github/repos/repo[@coords='%s']",
86                      this.coords
87                  )
88              ).addIf("forks")
89          );
90      }
91      @Override
92      public Repo repo() {
93          return new MkRepo(this.storage, this.self, this.coords);
94      }
95      /**
96       * Gets a mocked Fork.
97       * @param forkid Fork id
98       * @return Mocked Fork
99       */
100     public Fork get(final int forkid) {
101         return new MkFork(this.storage, forkid, this.coords);
102     }
103     @Override
104     public Iterable<Fork> iterate(
105         final String sort
106     ) {
107         return new MkIterable<>(
108             this.storage,
109             String.format("%s/fork", this.xpath()),
110             xml -> this.get(
111                 Integer.parseInt(xml.xpath("id/text()").get(0))
112             )
113         );
114     }
115     @Override
116     public Fork create(
117         final String org
118     ) throws IOException {
119         this.storage.lock();
120         final int number;
121         try {
122             number = 1 + this.storage.xml().xpath(
123                 String.format("%s/fork/id/text()", this.xpath())
124             ).size();
125             this.storage.apply(
126                 new Directives().xpath(this.xpath()).add("fork")
127                     .add("id").set(Integer.toString(number)).up()
128                     .attr("organization", org)
129             );
130         } finally {
131             this.storage.unlock();
132         }
133         Logger.info(
134             this, "fork %s created inside %s by %s",
135             this.coords, org, this.self
136         );
137         return this.get(number);
138     }
139 
140     /**
141      * XPath of this element in XML tree.
142      * @return XPath
143      */
144     private String xpath() {
145         return String.format(
146             "/github/repos/repo[@coords='%s']/forks",
147             this.coords
148         );
149     }
150 }
