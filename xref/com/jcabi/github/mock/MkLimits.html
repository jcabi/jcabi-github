5   package com.jcabi.github.mock;
6   
7   import com.jcabi.aspects.Immutable;
8   import com.jcabi.aspects.Loggable;
9   import com.jcabi.github.GitHub;
10  import com.jcabi.github.Limit;
11  import com.jcabi.github.Limits;
12  import jakarta.json.Json;
13  import jakarta.json.JsonObject;
14  import lombok.EqualsAndHashCode;
15  import lombok.ToString;
16  
17  /**
18   * Mock GitHub Rate Limit API.
19   *
20   * @since 0.6
21   */
22  @Immutable
23  @Loggable(Loggable.DEBUG)
24  @ToString
25  @EqualsAndHashCode(of = { "storage", "himself" })
26  final class MkLimits implements Limits {
27  
28      /**
29       * Storage.
30       */
31      private final transient MkStorage storage;
32  
33      /**
34       * Login of the user logged in.
35       */
36      private final transient String himself;
37  
38      /**
39       * Public ctor.
40       * @param stg Storage
41       * @param login User to login
42       */
43      MkLimits(final MkStorage stg,
44          final String login
45      ) {
46          this.storage = stg;
47          this.himself = login;
48      }
49  
50      @Override
51      public GitHub github() {
52          return new MkGitHub(this.storage, this.himself);
53      }
54  
55      @Override
56      public Limit get(final String resource) {
57          // @checkstyle AnonInnerLength (50 lines)
58          return new Limit() {
59              @Override
60              public GitHub github() {
61                  return MkLimits.this.github();
62              }
63  
64              @Override
65              public JsonObject json() {
66                  return Json.createObjectBuilder()
67                      // @checkstyle MagicNumber (2 lines)
68                      .add("limit", 5000)
69                      .add("remaining", 4999)
70                      .add("reset", System.currentTimeMillis())
71                      .build();
72              }
73          };
74      }
75  }
