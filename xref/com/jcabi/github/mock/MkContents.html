24  /**
25   * Mock GitHub contents.
26   *
27   * @since 0.8
28   */
29  @Immutable
30  @Loggable(Loggable.DEBUG)
31  @ToString
32  @EqualsAndHashCode(of = { "storage", "self", "coords" })
33  @SuppressWarnings({ "PMD.AvoidDuplicateLiterals", "PMD.TooManyMethods" })
34  final class MkContents implements Contents {
35  
36      /**
37       * Storage.
38       */
39      private final transient MkStorage storage;
40  
41      /**
42       * Login of the user logged in.
43       */
44      private final transient String self;
45  
46      /**
47       * Repo name.
48       */
49      private final transient Coordinates coords;
50  
51      /**
52       * Public ctor.
53       * @param stg Storage
54       * @param login User to login
55       * @param rep Repo
56       * @throws IOException If there is any I/O problem
57       */
58      @SuppressWarnings("PMD.ConstructorOnlyInitializesOrCallOtherConstructors")
59      MkContents(
60          final MkStorage stg,
61          final String login,
62          final Coordinates rep
63      ) throws IOException {
64          this.storage = stg;
65          this.self = login;
66          this.coords = rep;
67          this.storage.apply(
68              new Directives().xpath(
69                  String.format("/github/repos/repo[@coords='%s']", this.coords)
70              ).addIf("contents").up().addIf("commits")
71          );
72      }
73  
74      @Override
75      public Repo repo() {
76          return new MkRepo(this.storage, this.self, this.coords);
77      }
78  
79      @Override
80      public Content readme() throws IOException {
81          // @checkstyle MultipleStringLiterals (1 line)
82          return this.readme("master");
83      }
84  
85      @Override
86      public Content readme(
87          final String branch
88      ) {
89          return new MkContent(
90              this.storage, this.self, this.coords, "README.md", branch
91          );
92      }
93  
94      @Override
95      public Content create(
96          final JsonObject json
97      ) throws IOException {
98          this.storage.lock();
99          // @checkstyle MultipleStringLiterals (20 lines)
100         final String branch;
101         try {
102             if (json.containsKey("ref")) {
103                 branch = json.getString("ref");
104             } else {
105                 branch = "master";
106             }
107             this.storage.apply(
108                 new Directives().xpath(this.xpath()).add("content")
109                     .attr("ref", branch)
110                     .add("name").set(json.getString("path")).up()
111                     .add("path").set(json.getString("path")).up()
112                     .add("content").set(json.getString("content")).up()
113                     .add("type").set("file").up()
114                     .add("encoding").set("base64").up()
115                     .add("sha").set(MkContents.fakeSha()).up()
116                     .add("url").set("http://localhost/1").up()
117                     .add("git_url").set("http://localhost/2").up()
118                     .add("html_url").set("http://localhost/3").up()
119             );
120             this.commit(json);
121         } finally {
122             this.storage.unlock();
123         }
124         return new MkContent(
125             this.storage, this.self, this.coords, json.getString("path"), branch
126         );
127     }

135     }

138     public Content get(
139         final String path

143         );
144     }

146     @Override
147     public Iterable<Content> iterate(
148         final String pattern,
149         final String ref
150     ) throws IOException {
151         final Collection<XML> nodes = this.storage.xml().nodes(
152             this.xpath().concat(String.format("/content[@ref='%s']", ref))
153         );
154         final Collection<Content> result = new ArrayList<>(nodes.size());
155         for (final XML node : nodes) {
156             final String path = node.xpath("path/text()").get(0);
157             if (path.startsWith(pattern)) {
158                 result.add(
159                     this.mkContent(ref, path)
160                 );
161             }
162         }
163         return result;
164     }
165 
166     @Override
167     public RepoCommit remove(
168         final JsonObject content
169     ) throws IOException {
170         this.storage.lock();
171         final String path = content.getString("path");
172         // @checkstyle MultipleStringLiterals (20 lines)
173         try {
174             final String branch;
175             if (content.containsKey("ref")) {
176                 branch = content.getString("ref");
177             } else {
178                 branch = "master";
179             }
180             this.storage.apply(
181                 new Directives()
182                     .xpath(this.xpath())
183                     .xpath(String.format("content[path='%s']", path))
184                     .attr("ref", branch)
185                     .remove()
186             );
187             return this.commit(content);
188         } finally {
189             this.storage.unlock();
190         }
191     }
192 
193     @Override
194     public RepoCommit update(
195         final String path,
196         final JsonObject json
197     ) throws IOException {
198         this.storage.lock();
199         try {
200             final String ref = "ref";
201             final String branch;
202             if (json.containsKey(ref)) {
203                 branch = json.getString(ref);
204             } else {
205                 branch = "master";
206             }
207             final String xpath = String.format(
208                 // @checkstyle LineLengthCheck (1 line)
209                 "/github/repos/repo[@coords='%s']/contents/content[path='%s' and @ref='%s']",
210                 this.coords, path, branch
211             );
212             new JsonPatch(this.storage).patch(xpath, json);
213             return this.commit(json);
214         } finally {
215             this.storage.unlock();
216         }
217     }
218 
219     @Override
220     public boolean exists(final String path, final String ref)
221         throws IOException {
222         return this.storage.xml().nodes(
223             this.xpath().concat(String.format("/content[path='%s']", path))
224         ).size() > 0;
225     }

250      * @return Xpath

253         return String.format(
254             "/github/repos/repo[@coords='%s']/commits",
255             this.coords
256         );
257     }
258 
259     /**
260      * XML Directives for commit creation.
261      * @param json Source
262      * @return SHA string
263      * @throws IOException If an IO Exception occurs
264      */
265     private MkRepoCommit commit(
266         final JsonObject json
267     ) throws IOException {
268         final String sha = MkContents.fakeSha();
269         // @checkstyle MultipleStringLiterals (40 lines)
270         final Directives commit = new Directives().xpath(this.commitXpath())
271             .add("commit")
272             .add("sha").set(sha).up()
273             .add("url").set("http://localhost/4").up()
274             .add("html_url").set("http://localhost/5").up()
275             .add("message").set(json.getString("message")).up();
276         if (json.containsKey("committer")) {
277             final JsonObject committer = json.getJsonObject("committer");
278             commit.add("committer")
279                 .add("email").set(committer.getString("email")).up()
280                 .add("name").set(committer.getString("name")).up();
281         }
282         if (json.containsKey("author")) {
283             final JsonObject author = json.getJsonObject("author");
284             commit.add("author")
285                 .add("email").set(author.getString("email")).up()
286                 .add("name").set(author.getString("name")).up();
287         }
288         this.storage.apply(commit);
289         return new MkRepoCommit(this.storage, this.repo(), sha);
290     }

292     /**
293      * Generate a random fake SHA hex string.
294      *
295      * @return Fake SHA string.
296      */
297     private static String fakeSha() {
298         // @checkstyle MagicNumberCheck (1 line)
299         return RandomStringUtils.secure().next(40, "0123456789abcdef");
300     }
