29  /**
30   * Mock GitHub issue.

32   * @since 0.5

57       * Issue number.

61      /**

69      MkIssue(

118     @Override
119     public Iterable<Event> events() throws IOException {
120         return new MkIterable<>(
121             this.storage,

139     @Override

144     }

146     @Override
147     public void patch(
148         final JsonObject json
149     ) throws IOException {
150         final Issue.Smart smart = new Issue.Smart(this);
151         final boolean was = smart.isOpen();
152         new JsonPatch(this.storage).patch(this.xpath(), json);
153         final boolean now = smart.isOpen();
154         if (now != was) {
155             final String type;
156             if (now) {
157                 type = Event.REOPENED;
158             } else {
159                 type = Event.CLOSED;
160             }
161             new MkIssueEvents(this.storage, this.self, this.coords)
162                 .create(type, this.num, this.self, Optional.absent());
163         }
164     }
165 
166     @Override
167     public JsonObject json() throws IOException {
168         final XML xml = this.storage.xml();
169         final JsonObject obj = new JsonNode(
170             xml.nodes(this.xpath()).get(0)
171         ).json();
172         final JsonObjectBuilder json = Json.createObjectBuilder();
173         for (final Map.Entry<String, JsonValue> val: obj.entrySet()) {
174             json.add(val.getKey(), val.getValue());
175         }
176         final JsonArrayBuilder array = Json.createArrayBuilder();
177         for (final Label label : this.labels().iterate()) {
178             array.add(
179                 Json.createObjectBuilder().add("name", label.name()).build()
180             );
181         }
182         final JsonObjectBuilder res = json
183             .add("labels", array)
184             .add(
185                 // @checkstyle MultipleStringLiteralsCheck (1 line)
186                 "assignee",
187                 Json.createObjectBuilder().add(
188                     "login", obj.getString("assignee", "")
189                 ).build()
190             );
191         final JsonObjectBuilder pull = Json.createObjectBuilder();
192         final String html = "html_url";
193         if (xml.nodes(
194             String.format(
195                 // @checkstyle LineLengthCheck (1 line)
196                 "/github/repos/repo[@coords='%s']/pulls/pull/number[text() = '%d']",
197                 this.coords,
198                 this.num
199             )
200         ).isEmpty()) {
201             pull.addNull(html);
202         } else {
203             pull.add(
204                 html,
205                 String.format(
206                     "https://%s/pulls/%d",
207                     this.coords,
208                     this.num
209                 )
210             );
211         }
212         return res.add("pull_request", pull.build()).build();
213     }

251     /**
252      * Mapping for MkIssueEvents.
253      * @since 0.5
254      */
255     private static class MkIssueEventMapping
256         implements MkIterable.Mapping<Event> {
257         /**
258          * Issue events.
259          */
260         private final transient MkIssueEvents evts;
261 
262         /**
263          * Constructor.
264          * @param issue Mock issue to get events from
265          * @throws IOException If there is any I/O problem
266          */
267         MkIssueEventMapping(
268             final MkIssue issue
269         ) throws IOException {
270             this.evts = new MkIssueEvents(
271                 issue.storage,
272                 issue.self,
273                 issue.coords
274             );
275         }
276 
277         @Override
278         public Event map(
279             final XML xml
280         ) {
281             return this.evts.get(
282                 Integer.parseInt(xml.xpath("number/text()").get(0))
283             );
284         }
