5   package com.jcabi.github.mock;
6   
7   import com.jcabi.aspects.Immutable;
8   import com.jcabi.github.Coordinates;
9   import com.jcabi.github.Milestone;
10  import com.jcabi.github.Repo;
11  import jakarta.json.JsonObject;
12  import java.io.IOException;
13  
14  /**
15   * Mock GitHub milestone.
16   * @since 0.7
17   * @checkstyle MultipleStringLiterals (500 lines)
18   */
19  @Immutable
20  final class MkMilestone implements Milestone {
21  
22      /**
23       * Storage.
24       */
25      private final transient MkStorage storage;
26  
27      /**
28       * Login of the user logged in.
29       */
30      private final transient String self;
31  
32      /**
33       * Repo name.
34       */
35      private final transient Coordinates coords;
36  
37      /**
38       * The number of the MkMilestone.
39       */
40      private final transient int code;
41  
42      /**
43       * MkMilestone constructor.
44       * @param strg The storage
45       * @param login The user to login with
46       * @param crds The repo
47       * @param num The number of the MkMilestone
48       * @checkstyle ParameterNumber (5 lines)
49       */
50      MkMilestone(
51          final MkStorage strg,
52          final String login,
53          final Coordinates crds,
54          final int num
55      ) {
56          this.self = login;
57          this.coords = crds;
58          this.storage = strg;
59          this.code = num;
60      }
61  
62      @Override
63      public boolean equals(
64          final Object obj
65      ) {
66          return obj instanceof Milestone
67              && this.code == Milestone.class.cast(obj).number();
68      }
69  
70      @Override
71      public int hashCode() {
72          return this.code;
73      }
74  
75      @Override
76      public int compareTo(
77          final Milestone milestone
78      ) {
79          assert this.self != null;
80          assert this.coords != null;
81          assert this.storage != null;
82          assert this.code != -1;
83          int result = this.coords.compareTo(
84              milestone.repo().coordinates()
85          );
86          if (result == 0) {
87              result = Integer.compare(this.code, milestone.number());
88          }
89          return result;
90      }
91  
92      @Override
93      public JsonObject json() throws IOException {
94          return new JsonNode(
95              this.storage.xml().nodes(this.xpath()).get(0)
96          ).json();
97      }
98  
99      @Override
100     public void patch(
101         final JsonObject json
102     ) throws IOException {
103         new JsonPatch(
104             this.storage
105         ).patch(
106             this.xpath(),
107             json
108         );
109     }
110 
111     @Override
112     public Repo repo() {
113         return new MkRepo(
114             this.storage,
115             this.self,
116             this.coords
117         );
118     }
119 
120     @Override
121     public int number() {
122         return this.code;
123     }
124 
125     /**
126      * XPath of this element in XML tree.
127      * @return XPath
128      */
129     private String xpath() {
130         return String.format(
131             "/github/repos/repo[@coords='%s']/milestones[number='%d']",
132             this.coords,
133             this.code
134         );
135     }
136 
137 }
