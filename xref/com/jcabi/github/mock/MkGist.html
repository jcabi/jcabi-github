5   package com.jcabi.github.mock;
6   
7   import com.jcabi.aspects.Immutable;
8   import com.jcabi.aspects.Loggable;
9   import com.jcabi.github.Gist;
10  import com.jcabi.github.GistComments;
11  import com.jcabi.github.GitHub;
12  import com.jcabi.xml.XML;
13  import jakarta.json.JsonObject;
14  import java.io.IOException;
15  import java.util.List;
16  import lombok.ToString;
17  import org.xembly.Directives;
18  
19  /**
20   * Mock GitHub gist.
21   *
22   * @since 0.5
23   */
24  @Immutable
25  @Loggable(Loggable.DEBUG)
26  @ToString
27  @SuppressWarnings("PMD.TooManyMethods")
28  final class MkGist implements Gist {
29  
30      /**
31       * XPath for gist IDs.
32       */
33      private static final String GIST_ID_XPATH = "/github/gists/gist/id/text()";
34  
35      /**
36       * Storage.
37       */
38      private final transient MkStorage storage;
39  
40      /**
41       * Login of the user logged in.
42       */
43      private final transient String self;
44  
45      /**
46       * Gist name.
47       */
48      private final transient String gist;
49  
50      /**
51       * Public ctor.
52       * @param stg Storage
53       * @param login User to login
54       * @param name Gist name
55       * @checkstyle ParameterNumber (5 lines)
56       */
57      MkGist(
58          final MkStorage stg,
59          final String login,
60          final String name
61      ) {
62          this.storage = stg;
63          this.self = login;
64          this.gist = name;
65      }
66  
67      @Override
68      public GitHub github() {
69          return new MkGitHub(this.storage, this.self);
70      }
71  
72      @Override
73      public String identifier() {
74          return this.gist;
75      }
76  
77      @Override
78      public String read(
79          final String file
80      ) throws IOException {
81          final List<XML> files = this.storage.xml().nodes(
82              String.format(
83                  "%s/files/file[filename='%s']",
84                  this.xpath(), file
85              )
86          );
87          if (files.isEmpty()) {
88              throw new IOException(
89                  String.format("Couldn't find file with the name %s.", file)
90              );
91          }
92          final List<String> contents = files.get(0)
93              .xpath("raw_content/text()");
94          String content = "";
95          if (!contents.isEmpty()) {
96              content = contents.get(0);
97          }
98          return content;
99      }
100 
101     @Override
102     public void write(
103         final String file,
104         final String content
105     )
106         throws IOException {
107         this.storage.apply(
108             // @checkstyle MultipleStringLiterals (3 lines)
109             new Directives().xpath(this.xpath()).xpath(
110                 String.format("files[not(file[filename='%s'])]", file)
111             ).add("file").add("filename").set(file).up().add("raw_content")
112         );
113         this.storage.apply(
114             new Directives().xpath(this.xpath()).xpath(
115                 String.format(
116                     "files/file[filename='%s']/raw_content",
117                     file
118                 )
119             ).set(content)
120         );
121     }
122 
123     @Override
124     public void star() throws IOException {
125         this.storage.apply(
126             new Directives()
127                 .xpath(this.xpath())
128                 .attr("starred", Boolean.toString(true))
129         );
130     }
131 
132     @Override
133     public void unstar() throws IOException {
134         this.storage.apply(
135             new Directives()
136                 .xpath(this.xpath())
137                 .attr("starred", Boolean.toString(false))
138         );
139     }
140 
141     @Override
142     public boolean starred() throws IOException {
143         final List<String> xpath = this.storage.xml().xpath(
144             this.xpath().concat("/@starred")
145         );
146         return !xpath.isEmpty()
147             && Boolean.toString(true).equalsIgnoreCase(xpath.get(0));
148     }
149 
150     @Override
151     public Gist fork() throws IOException {
152         this.storage.lock();
153         final String number;
154         try {
155             final XML xml = this.storage.xml();
156             number = Integer.toString(
157                 1 + xml.xpath(MkGist.GIST_ID_XPATH).size()
158             );
159             final Directives dirs = new Directives().xpath("/github/gists")
160                 .add("gist")
161                 .add("id").set(number).up()
162                 .add("files");
163             final List<XML> files = xml.nodes(
164                 this.xpath().concat("/files/file")
165             );
166             for (final XML file : files) {
167                 final String filename = file.xpath("filename/text()").get(0);
168                 // @checkstyle MultipleStringLiterals (3 lines)
169                 dirs.add("file")
170                     .add("filename").set(filename).up()
171                     .add("raw_content").set(this.read(filename)).up().up();
172             }
173             this.storage.apply(dirs);
174         } finally {
175             this.storage.unlock();
176         }
177         return new MkGist(this.storage, this.self, number);
178     }
179 
180     @Override
181     public GistComments comments() {
182         throw new UnsupportedOperationException();
183     }
184 
185     @Override
186     public JsonObject json() throws IOException {
187         return new JsonNode(
188             this.storage.xml().nodes(this.xpath()).get(0)
189         ).json();
190     }
191 
192     @Override
193     public void patch(
194         final JsonObject json
195     ) throws IOException {
196         new JsonPatch(this.storage).patch(this.xpath(), json);
197     }
198 
199     @Override
200     public boolean equals(final Object obj) {
201         final boolean result;
202         if (this == obj) {
203             result = true;
204         } else if (obj == null || this.getClass() != obj.getClass()) {
205             result = false;
206         } else {
207             final MkGist other = (MkGist) obj;
208             result = this.storage.equals(other.storage)
209                 && this.self.equals(other.self)
210                 && this.gist.equals(other.gist);
211         }
212         return result;
213     }
214 
215     @Override
216     public int hashCode() {
217         int result = this.storage.hashCode();
218         result = 31 * result + this.self.hashCode();
219         result = 31 * result + this.gist.hashCode();
220         return result;
221     }
222 
223     /**
224      * XPath of this element in XML tree.
225      * @return XPath
226      */
227     private String xpath() {
228         return String.format(
229             "/github/gists/gist[id='%s']",
230             this.gist
231         );
232     }
233 
234 }
