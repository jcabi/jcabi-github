5   package com.jcabi.github.mock;
6   
7   import com.google.common.base.Optional;
8   import com.jcabi.aspects.Immutable;
9   import com.jcabi.aspects.Loggable;
10  import com.jcabi.github.Coordinates;
11  import com.jcabi.github.Event;
12  import com.jcabi.github.Issue;
13  import com.jcabi.github.IssueLabels;
14  import com.jcabi.github.Label;
15  import java.io.IOException;
16  import java.util.Collection;
17  import java.util.HashSet;
18  import java.util.Set;
19  import lombok.EqualsAndHashCode;
20  import lombok.ToString;
21  import org.xembly.Directives;
22  
23  /**
24   * Mock GitHub labels.
25   *
26   * @since 0.5
27   */
28  @Immutable
29  @Loggable(Loggable.DEBUG)
30  @ToString
31  @EqualsAndHashCode(of = { "storage", "repo", "ticket" })
32  @SuppressWarnings("PMD.ConstructorOnlyInitializesOrCallOtherConstructors")
33  final class MkIssueLabels implements IssueLabels {
34  
35      /**
36       * Storage.
37       */
38      private final transient MkStorage storage;
39  
40      /**
41       * Login of the user logged in.
42       */
43      private final transient String self;
44  
45      /**
46       * Repo name.
47       */
48      private final transient Coordinates repo;
49  
50      /**
51       * Issue number.
52       */
53      private final transient int ticket;
54  
55      /**
56       * Public ctor.
57       * @param stg Storage
58       * @param login Login
59       * @param rep Repo
60       * @param issue Issue number
61       * @throws IOException If fails
62       * @checkstyle ParameterNumber (5 lines)
63       */
64      MkIssueLabels(
65          final MkStorage stg,
66          final String login,
67          final Coordinates rep,
68          final int issue
69      ) throws IOException {
70          this.storage = stg;
71          this.self = login;
72          this.repo = rep;
73          this.ticket = issue;
74          this.storage.apply(
75              new Directives().xpath(
76                  String.format(
77                      // @checkstyle LineLength (1 line)
78                      "/github/repos/repo[@coords='%s']/issues/issue[number='%d']",
79                      rep, this.ticket
80                  )
81              ).addIf("labels")
82          );
83      }
84  
85      @Override
86      public Issue issue() {
87          return new MkIssue(this.storage, this.self, this.repo, this.ticket);
88      }
89  
90      @Override
91      public void add(final Iterable<String> labels
92      ) throws IOException {
93          final Collection<String> existing = this.labels();
94          final Set<String> added = new HashSet<>();
95          final Directives dirs = new Directives().xpath(this.xpath());
96          for (final String label : labels) {
97              dirs.add("label").set(label).up();
98              if (!existing.contains(label)) {
99                  added.add(label);
100             }
101         }
102         this.storage.apply(dirs);
103         if (!added.isEmpty()) {
104             final MkIssueEvents events = new MkIssueEvents(
105                 this.storage,
106                 this.self,
107                 this.repo
108             );
109             for (final String label : added) {
110                 events.create(
111                     Event.LABELED,
112                     this.ticket,
113                     this.self,
114                     Optional.of(label)
115                 );
116             }
117         }
118     }
119 
120     @Override
121     public void replace(final Iterable<String> labels
122     ) throws IOException {
123         this.clear();
124         this.add(labels);
125     }
126 
127     @Override
128     public Iterable<Label> iterate() {
129         return new MkIterable<>(
130             this.storage,
131             this.xpath().concat("/*"),
132             xml -> new MkLabel(
133                 this.storage,
134                 this.self,
135                 this.repo,
136                 xml.xpath("./text()").get(0)
137             )
138         );
139     }
140 
141     @Override
142     public void remove(final String name
143     ) throws IOException {
144         if (this.labels().contains(name)) {
145             this.storage.apply(
146                 new Directives().xpath(
147                     this.xpath().concat(String.format("/label[.='%s']", name))
148                 ).remove()
149             );
150             new MkIssueEvents(
151                 this.storage,
152                 this.self,
153                 this.repo
154             ).create(
155                 Event.UNLABELED,
156                 this.ticket,
157                 this.self,
158                 Optional.of(name)
159             );
160         }
161     }
162 
163     @Override
164     public void clear() throws IOException {
165         for (final String label : this.labels()) {
166             this.remove(label);
167         }
168     }
169 
170     /**
171      * XPath of this element in XML tree.
172      * @return XPath
173      */
174     private String xpath() {
175         return String.format(
176             "/github/repos/repo[@coords='%s']/issues/issue[number='%d']/labels",
177             this.repo, this.ticket
178         );
179     }
180 
181     /**
182      * Returns a set of all of the issue's labels.
183      * @return Set of label names
184      */
185     private Collection<String> labels() {
186         final Set<String> labels = new HashSet<>();
187         for (final Label label : this.iterate()) {
188             labels.add(label.name());
189         }
190         return labels;
191     }
192 }
