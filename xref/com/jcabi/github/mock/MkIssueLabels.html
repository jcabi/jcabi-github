30  package com.jcabi.github.mock;
31  
32  import com.google.common.base.Optional;
33  import com.jcabi.aspects.Immutable;
34  import com.jcabi.aspects.Loggable;
35  import com.jcabi.github.Coordinates;
36  import com.jcabi.github.Event;
37  import com.jcabi.github.Issue;
38  import com.jcabi.github.IssueLabels;
39  import com.jcabi.github.Label;
40  import java.io.IOException;
41  import java.util.Collection;
42  import java.util.HashSet;
43  import java.util.Set;
44  import lombok.EqualsAndHashCode;
45  import lombok.ToString;
46  import org.xembly.Directives;
47  
48  /**
49   * Mock Github labels.
50   *
51   * @author Yegor Bugayenko (yegor256@gmail.com)
52   * @version $Id: 2888e4f42fed0306e25bae434da3d990f4e83d06 $
53   * @since 0.5
54   */
55  @Immutable
56  @Loggable(Loggable.DEBUG)
57  @ToString
58  @EqualsAndHashCode(of = { "storage", "repo", "ticket" })
59  final class MkIssueLabels implements IssueLabels {
60  
61      /**
62       * Storage.
63       */
64      private final transient MkStorage storage;
65  
66      /**
67       * Login of the user logged in.
68       */
69      private final transient String self;
70  
71      /**
72       * Repo name.
73       */
74      private final transient Coordinates repo;
75  
76      /**
77       * Issue number.
78       */
79      private final transient int ticket;
80  
81      /**
82       * Public ctor.
83       * @param stg Storage
84       * @param login Login
85       * @param rep Repo
86       * @param issue Issue number
87       * @throws IOException If fails
88       * @checkstyle ParameterNumber (5 lines)
89       */
90      MkIssueLabels(
91          final MkStorage stg,
92          final String login,
93          final Coordinates rep,
94          final int issue
95      ) throws IOException {
96          this.storage = stg;
97          this.self = login;
98          this.repo = rep;
99          this.ticket = issue;
100         this.storage.apply(
101             new Directives().xpath(
102                 String.format(
103                     // @checkstyle LineLength (1 line)
104                     "/github/repos/repo[@coords='%s']/issues/issue[number='%d']",
105                     rep, this.ticket
106                 )
107             ).addIf("labels")
108         );
109     }
110 
111     @Override
112     public Issue issue() {
113         return new MkIssue(this.storage, this.self, this.repo, this.ticket);
114     }
115 
116     @Override
117     public void add(final Iterable<String> labels
118     ) throws IOException {
119         final Collection<String> existing = this.labels();
120         final Set<String> added = new HashSet<>();
121         final Directives dirs = new Directives().xpath(this.xpath());
122         for (final String label : labels) {
123             dirs.add("label").set(label).up();
124             if (!existing.contains(label)) {
125                 added.add(label);
126             }
127         }
128         this.storage.apply(dirs);
129         if (!added.isEmpty()) {
130             final MkIssueEvents events = new MkIssueEvents(
131                 this.storage,
132                 this.self,
133                 this.repo
134             );
135             for (final String label : added) {
136                 events.create(
137                     Event.LABELED,
138                     this.ticket,
139                     this.self,
140                     Optional.of(label)
141                 );
142             }
143         }
144     }
145 
146     @Override
147     public void replace(final Iterable<String> labels
148     ) throws IOException {
149         this.clear();
150         this.add(labels);
151     }
152 
153     @Override
154     public Iterable<Label> iterate() {
155         return new MkIterable<>(
156             this.storage,
157             String.format("%s/*", this.xpath()),
158             xml -> new MkLabel(
159                 this.storage,
160                 this.self,
161                 this.repo,
162                 xml.xpath("./text()").get(0)
163             )
164         );
165     }
166 
167     @Override
168     public void remove(final String name
169     ) throws IOException {
170         if (this.labels().contains(name)) {
171             this.storage.apply(
172                 new Directives().xpath(
173                     String.format("%s/label[.='%s']", this.xpath(), name)
174                 ).remove()
175             );
176             new MkIssueEvents(
177                 this.storage,
178                 this.self,
179                 this.repo
180             ).create(
181                 Event.UNLABELED,
182                 this.ticket,
183                 this.self,
184                 Optional.of(name)
185             );
186         }
187     }
188 
189     @Override
190     public void clear() throws IOException {
191         for (final String label : this.labels()) {
192             this.remove(label);
193         }
194     }
195 
196     /**
197      * XPath of this element in XML tree.
198      * @return XPath
199      */
200     private String xpath() {
201         return String.format(
202             "/github/repos/repo[@coords='%s']/issues/issue[number='%d']/labels",
203             this.repo, this.ticket
204         );
205     }
206 
207     /**
208      * Returns a set of all of the issue's labels.
209      * @return Set of label names
210      */
211     private Collection<String> labels() {
212         final Set<String> labels = new HashSet<>();
213         for (final Label label : this.iterate()) {
214             labels.add(label.name());
215         }
216         return labels;
217     }
218 }
