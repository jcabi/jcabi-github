5   package com.jcabi.github.mock;
6   
7   import com.jcabi.aspects.Immutable;
8   import com.jcabi.aspects.Loggable;
9   import com.jcabi.github.Commit;
10  import com.jcabi.github.Commits;
11  import com.jcabi.github.Coordinates;
12  import com.jcabi.github.Repo;
13  import com.jcabi.github.Statuses;
14  import jakarta.json.JsonObject;
15  import java.io.IOException;
16  import lombok.EqualsAndHashCode;
17  import org.xembly.Directives;
18  
19  /**
20   * Mock of GitHub Commits.
21   * @since 0.3
22   */
23  @Immutable
24  @Loggable(Loggable.DEBUG)
25  @EqualsAndHashCode(of = { "storage", "self", "coords" })
26  @SuppressWarnings("PMD.ConstructorOnlyInitializesOrCallOtherConstructors")
27  public final class MkCommits implements Commits {
28  
29      /**
30       * Storage.
31       */
32      private final transient MkStorage storage;
33  
34      /**
35       * Login of the user logged in.
36       */
37      private final transient String self;
38  
39      /**
40       * Repo's name.
41       */
42      private final transient Coordinates coords;
43  
44      /**
45       * Public constructor.
46       * @param stg The storage.
47       * @param login The login name.
48       * @param rep Repo's coordinates.
49       * @throws IOException If something goes wrong.
50       */
51      MkCommits(
52          final MkStorage stg,
53          final String login,
54          final Coordinates rep
55      ) throws IOException {
56          this.storage = stg;
57          this.self = login;
58          this.coords = rep;
59          this.storage.apply(
60              new Directives().xpath(
61                  String.format(
62                      "/github/repos/repo[@coords='%s']/git",
63                      this.coords
64                  )
65              ).addIf("commits")
66          );
67      }
68  
69      @Override
70      public Repo repo() {
71          return new MkRepo(this.storage, this.self, this.coords);
72      }
73  
74      @Override
75      public Commit create(
76          final JsonObject params
77      ) {
78          return this.get(params.getString("sha"));
79      }
80  
81      @Override
82      public Commit get(
83          final String sha
84      ) {
85          return new MkCommit(this.storage, this.self, this.coords, sha);
86      }
87  
88      @Override
89      public Statuses statuses(
90          final String sha
91      ) {
92          return new MkStatuses(this.get(sha));
93      }
94  }
