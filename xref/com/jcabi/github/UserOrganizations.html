5   package com.jcabi.github;
6   
7   import com.jcabi.aspects.Immutable;
8   import java.io.IOException;
9   
10  /**
11   * Organizations of a GitHub user.
12   * @see <a href="https://developer.github.com/v3/orgs/">Organizations API</a>
13   * @since 0.24
14   */
15  @Immutable
16  public interface UserOrganizations {
17  
18      /**
19       * GitHub we're in.
20       * @return GitHub
21       */
22      GitHub github();
23  
24      /**
25       * Get its owner.
26       * @return User
27       */
28      User user();
29  
30      /**
31       * Iterate organizations of particular user.
32       * All public organizations for an unauthenticated user or
33       * private and public organizations for authenticated users
34       * @return Iterator of Organizations
35       * @throws IOException If there is an I/O problem
36       * @see <a href="https://developer.github.com/v3/orgs/#list-user-organizations">List User Organizations</a>
37       * @since 0.24
38       */
39      Iterable<Organization> iterate() throws IOException;
40  }
