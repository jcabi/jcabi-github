30  package com.jcabi.github;
31  
32  import com.jcabi.aspects.Tv;
33  import com.jcabi.github.OAuthScope.Scope;
34  import org.apache.commons.lang3.RandomStringUtils;
35  import org.hamcrest.MatcherAssert;
36  import org.hamcrest.Matchers;
37  import org.junit.AfterClass;
38  import org.junit.Assume;
39  import org.junit.BeforeClass;
40  import org.junit.Test;
41  
42  /**
43   * Test case for {@link RtReleases}.
44   * @author Paul Polishchuk (ppol@ua.fm)
45   * @version $Id: 74089297b82c0342a574441195de53ea4af36a0f $
46   * @since 0.8
47   */
48  @OAuthScope(Scope.REPO)
49  public final class RtReleasesITCase {
50  
51      /**
52       * Test repos.
53       */
54      private static Repos repos;
55  
56      /**
57       * Test repo.
58       */
59      private static Repo repo;
60  
61      /**
62       * RepoRule.
63       * @checkstyle VisibilityModifierCheck (3 lines)
64       */
65      private static RepoRule rule = new RepoRule();
66  
67      /**
68       * Set up test fixtures.
69       * @throws Exception If some errors occurred.
70       */
71      @BeforeClass
72      public static void setUp() throws Exception {
73          final String key = System.getProperty("failsafe.github.key");
74          Assume.assumeThat(key, Matchers.notNullValue());
75          final Github github = new RtGithub(key);
76          repos = github.repos();
77          repo = rule.repo(repos);
78      }
79  
80      /**
81       * Tear down test fixtures.
82       * @throws Exception If some errors occurred.
83       */
84      @AfterClass
85      public static void tearDown() throws Exception {
86          if (repos != null && repo != null) {
87              repos.remove(repo.coordinates());
88          }
89      }
90  
91      /**
92       * RtReleases can iterate releases.
93       * @throws Exception if something goes wrong
94       */
95      @Test
96      public void canFetchAllReleases() throws Exception {
97          final Releases releases = repo.releases();
98          final Release release = releases.create(
99              RandomStringUtils.randomAlphanumeric(Tv.TEN)
100         );
101         try {
102             MatcherAssert.assertThat(
103                 releases.iterate(),
104                 Matchers.not(Matchers.emptyIterableOf(Release.class))
105             );
106         } finally {
107             releases.remove(release.number());
108         }
109     }
110 
111     /**
112      * RtReleases can fetch a single release.
113      * @throws Exception if any error inside
114      */
115     @Test
116     public void canFetchRelease() throws Exception {
117         final Releases releases = repo.releases();
118         final String tag = "v1.0";
119         final Release release = releases.create(tag);
120         MatcherAssert.assertThat(
121             releases.get(release.number()).number(),
122             Matchers.equalTo(release.number())
123         );
124         MatcherAssert.assertThat(
125             new Release.Smart(releases.get(release.number())).tag(),
126             Matchers.equalTo(tag)
127         );
128         releases.remove(release.number());
129     }
130 
131     /**
132      * RtReleases can create a release.
133      * @throws Exception if any error inside
134      */
135     @Test
136     public void canCreateRelease() throws Exception {
137         final Releases releases = repo.releases();
138         final Release created = releases.create("0.1");
139         final int number = created.number();
140         try {
141             final Release obtained = releases.get(number);
142             MatcherAssert.assertThat(
143                 created,
144                 Matchers.is(obtained)
145             );
146             MatcherAssert.assertThat(
147                 new Release.Smart(created).tag(),
148                 Matchers.equalTo(new Release.Smart(obtained).tag())
149             );
150         } finally {
151             releases.remove(number);
152         }
153     }
154 
155     /**
156      * RtReleases can remove a release.
157      * @throws Exception if any problem inside
158      */
159     @Test
160     public void canRemoveRelease() throws Exception {
161         final Releases releases = repo.releases();
162         final Release release = releases.create(
163             RandomStringUtils.randomAlphanumeric(Tv.TEN)
164         );
165         MatcherAssert.assertThat(
166             releases.iterate(),
167             Matchers.hasItem(release)
168         );
169         releases.remove(release.number());
170         MatcherAssert.assertThat(
171             releases.iterate(),
172             Matchers.not(Matchers.hasItem(release))
173         );
174     }
175 
176     /**
177      * RtReleases can edit release tag.
178      * @throws Exception if any problem inside.
179      */
180     @Test
181     public void canEditTag() throws Exception {
182         final Releases releases = repo.releases();
183         final Release release = releases.create(
184             RandomStringUtils.randomAlphanumeric(Tv.TEN)
185         );
186         final String tag = RandomStringUtils.randomAlphanumeric(Tv.FIFTEEN);
187         new Release.Smart(release).tag(tag);
188         MatcherAssert.assertThat(
189             new Release.Smart(releases.get(release.number())).tag(),
190             Matchers.equalTo(tag)
191         );
192         releases.remove(release.number());
193     }
194 
195     /**
196      * RtReleases can edit release body.
197      * @throws Exception if any problem inside.
198      */
199     @Test
200     public void canEditBody() throws Exception {
201         final Releases releases = repo.releases();
202         final Release release = releases.create(
203             RandomStringUtils.randomAlphanumeric(Tv.TEN)
204         );
205         final String body = "Description of the release";
206         new Release.Smart(release).body(body);
207         MatcherAssert.assertThat(
208             new Release.Smart(releases.get(release.number())).body(),
209             Matchers.equalTo(body)
210         );
211         releases.remove(release.number());
212     }
213 }
