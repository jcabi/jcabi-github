30  package com.jcabi.github.mock;
31  
32  import com.jcabi.github.Gitignores;
33  import org.hamcrest.MatcherAssert;
34  import org.hamcrest.Matchers;
35  import org.junit.Test;
36  
37  /**
38   * Test case for {@link com.jcabi.github.mock.MkGitignores}.
39   * @author Paul Polishchuk (ppol@ua.fm)
40   * @version $Id: d6947fd20725d57ab74a61c458803607faefa573 $
41   * @since 0.8
42   */
43  public final class MkGitignoresTest {
44      /**
45       * MkGitignores can fetch single gitignore template.
46       * @throws Exception if some problem inside
47       */
48      @Test
49      public void canFetchSingleRawTemplate() throws Exception {
50          final Gitignores gitignores = new MkGithub().gitignores();
51          MatcherAssert.assertThat(
52              gitignores.template("Java"),
53              Matchers.startsWith("*.class")
54          );
55      }
56  
57      /**
58       * MkGitignores can iterate over templates.
59       * @throws Exception if something goes wrong.
60       */
61      @Test
62      public void canIterateOverTemplates() throws Exception {
63          final Gitignores gitignores = new MkGithub().gitignores();
64          MatcherAssert.assertThat(
65              gitignores.iterate(),
66              Matchers.not(Matchers.emptyIterable())
67          );
68      }
69  }
