30  package com.jcabi.github.mock;
31  
32  import com.jcabi.xml.XML;
33  import javax.json.Json;
34  import org.hamcrest.MatcherAssert;
35  import org.hamcrest.Matchers;
36  import org.junit.Test;
37  
38  /**
39   * Test case for {@link JsonPatch}.
40   * @author Yegor Bugayenko (yegor256@gmail.com)
41   * @version $Id: 9c9ec19df4f811a2368a85cd831317c414765490 $
42   */
43  public final class JsonPatchTest {
44  
45      /**
46       * JsonPatch can patch an XML.
47       * @throws Exception If some problem inside
48       */
49      @Test
50      public void patchesXml() throws Exception {
51          final MkStorage storage = new MkStorage.InFile();
52          new JsonPatch(storage).patch(
53              "/github",
54              Json.createObjectBuilder()
55                  .add("name", "hi you!")
56                  .add("number", 1)
57                  .build()
58          );
59          final XML xml = storage.xml();
60          MatcherAssert.assertThat(
61              xml.xpath("/github/name/text()").get(0),
62              Matchers.describedAs(xml.toString(), Matchers.endsWith("you!"))
63          );
64      }
65  
66  }
