45  /**

51  @SuppressWarnings("PMD.TooManyMethods")

54      /**
55       * MkIssue can open and close.
56       * @throws Exception If some problem inside
57       */
58      @Test
59      public void opensAndCloses() throws Exception {
60          final Issue issue = this.issue();
61          MatcherAssert.assertThat(
62              new Issue.Smart(issue).isOpen(),
63              Matchers.is(true)
64          );
65          new Issue.Smart(issue).close();
66          MatcherAssert.assertThat(
67              new Issue.Smart(issue).isOpen(),
68              Matchers.is(false)
69          );
70      }

73       * MkIssue can point to an absent pull request.
74       * @throws Exception If some problem inside

76      @Test
77      public void pointsToAnEmptyPullRequest() throws Exception {

79          MatcherAssert.assertThat(
80              new Issue.Smart(issue).isPull(),
81              Matchers.is(false)
82          );
83      }

89      @Test

138     /**
139      * MkIssue can list its labels.
140      * @throws Exception If some problem inside
141      */
142     @Test
143     public void listsReadOnlyLabels() throws Exception {
144         final Issue issue = this.issue();
145         final String tag = "test-tag";
146         issue.repo().labels().create(tag, "c0c0c0");
147         issue.labels().add(Collections.singletonList(tag));
148         MatcherAssert.assertThat(
149             new Issue.Smart(issue).roLabels().iterate(),
150             Matchers.<Label>hasItem(
151                 new CustomMatcher<Label>("label just created") {
152                     @Override
153                     public boolean matches(final Object item) {
154                         return Label.class.cast(item).name().equals(tag);
155                     }
156                 }
157             )
158         );
159     }

161     /**
162      * MkIssue should be able to compare different instances.
163      *
164      * @throws Exception when a problem occurs.
165      */
166     @Test
167     public void canCompareInstances() throws Exception {
168         final MkIssue less = new MkIssue(
169             new MkStorage.InFile(),
170             "login-less",
171             Mockito.mock(Coordinates.class),
172             1
173         );
174         final MkIssue greater = new MkIssue(
175             new MkStorage.InFile(),
176             "login-greater",
177             Mockito.mock(Coordinates.class),
178             2
179         );
180         MatcherAssert.assertThat(
181             less.compareTo(greater),
182             Matchers.lessThan(0)
183         );
184         MatcherAssert.assertThat(
185             greater.compareTo(less),
186             Matchers.greaterThan(0)
187         );
188     }
189 
190     /**
191      * MkIssue can remember it's author.
192      *
193      * @throws Exception when a problem occurs.
194      */
195     @Test
196     public void canRememberItsAuthor() throws Exception {
197         final MkGithub first = new MkGithub("first");
198         final Github second = first.relogin("second");
199         final Repo repo = first.repos().create(
200             Json.createObjectBuilder().add("name", "test").build()
201         );
202         final int number = second.repos()
203             .get(repo.coordinates())
204             .issues()
205             .create("", "")
206             .number();
207         final Issue issue = first.repos()
208             .get(repo.coordinates())
209             .issues()
210             .get(number);
211         MatcherAssert.assertThat(
212             new Issue.Smart(issue).author().login(),
213             Matchers.is("second")
214         );
215     }
