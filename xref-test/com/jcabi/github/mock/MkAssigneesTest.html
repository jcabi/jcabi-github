30  package com.jcabi.github.mock;
31  
32  import com.jcabi.github.Repo;
33  import com.jcabi.github.User;
34  import org.hamcrest.MatcherAssert;
35  import org.hamcrest.Matchers;
36  import org.junit.Test;
37  
38  /**
39   * Test case for {@link MkAssignees}.
40   * @author Paul Polishchuk (ppol@ua.fm)
41   * @version $Id: 67edc901309e7417458975b77fdefa883fdebe38 $
42   * @since 0.7
43   * @checkstyle MultipleStringLiteralsCheck (100 lines)
44   */
45  public final class MkAssigneesTest {
46  
47      /**
48       * MkAssignees can iterate over assignees.
49       * @throws Exception Exception If some problem inside
50       */
51      @Test
52      public void iteratesAssignees() throws Exception {
53          MatcherAssert.assertThat(
54              repo().assignees().iterate(),
55              Matchers.not(Matchers.emptyIterableOf(User.class))
56          );
57      }
58  
59      /**
60       * MkAssignees can check if a collaborator is an assignee for this repo.
61       * @throws Exception Exception If some problem inside
62       */
63      @Test
64      public void checkCollaboratorIsAssigneeForRepo() throws Exception {
65          final Repo repo = repo();
66          repo.collaborators().add("Vladimir");
67          MatcherAssert.assertThat(
68              repo.assignees().check("Vladimir"),
69              Matchers.is(true)
70          );
71      }
72  
73      /**
74       * MkAssignees can check if the owner is an assignee for this repo.
75       * @throws Exception Exception If some problem inside
76       */
77      @Test
78      public void checkOwnerIsAssigneeForRepo() throws Exception {
79          MatcherAssert.assertThat(
80              repo().assignees().check("Jonathan"),
81              Matchers.is(true)
82          );
83      }
84  
85      /**
86       * Create a repo to work with.
87       * @return Repo
88       * @throws Exception If some problem inside
89       */
90      private static Repo repo() throws Exception {
91          return new MkGithub("Jonathan").randomRepo();
92      }
93  }
