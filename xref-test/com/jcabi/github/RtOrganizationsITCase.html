32  import org.hamcrest.MatcherAssert;
33  import org.hamcrest.Matchers;
34  import org.junit.Test;
35  
36  /**
37   * Test case for {@link RtOrganizations}.
38   * @author Paul Polishchuk (ppol@ua.fm)
39   * @version $Id: 7474f363213085b7524a4ac0b0b1bc07ada2839a $
40   * @see <a href="http://developer.github.com/v3/orgs/">Organizations API</a>
41   * @since 0.24
42   */
43  public final class RtOrganizationsITCase {
44      /**
45       * RtOrganizations can get an organization.
46       * @throws Exception if any problem inside
47       */
48      @Test
49      public void getOrganization() throws Exception {
50          final String login = "github";
51          final Organization org = new GithubIT().connect()
52              .organizations().get(login);
53          MatcherAssert.assertThat(org.login(), Matchers.equalTo(login));
54      }

56  }
