53      /**
54       * Method 'iterate()' returns empty iterable if the service responds with
55       * no notifications.
56       */
57      @Test
58      public void iterateEmpty() {
59          MatcherAssert.assertThat(
60              new RtNotifications(
61                  new FakeRequest()
62                      .withBody("[]")
63              ).iterate(),
64              Matchers.emptyIterable()
65          );
66      }
